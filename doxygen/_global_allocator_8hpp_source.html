154 template< typename T, Core MASTER_CORE = 0 >
155 GlobalAddress<T> symmetric_global_alloc() {
156  static_assert(sizeof(T) % block_size == 0,
157  "must pad global proxy to multiple of block_size, or use GRAPPA_BLOCK_ALIGNED");
158  // allocate enough space that we are guaranteed to get one on each core at same location
159  auto qac = global_alloc<char>(cores()*(sizeof(T)+block_size));
160  while (qac.core() != MASTER_CORE) qac++;
161  auto qa = static_cast<GlobalAddress<T>>(qac);
162  CHECK_EQ(qa, qa.block_min());
163  CHECK_EQ(qa.core(), MASTER_CORE);
164  return qa;
165 }
