114  static void master_combine(GlobalAddress<GlobalVector> self) {
115  auto* m = &self->master;
116  m->combining = true;
117 
118  // TODO: make me less ugly...
119  if (FLAGS_flat_combining_local_only) {
120  if (m->has_requests()) {
121  m->ce.enroll();
122  enum class Choice { POP, DEQ, PUSH };
123  Choice c;
124  if (!m->push_q.empty()) {
125  c = Choice::PUSH;
126  invoke(m->push_q.pop());
127  } else if (!m->pop_q.empty()) {
128  c = Choice::POP;
129  invoke(m->pop_q.pop());
130  } else if (!m->deq_q.empty()) {
131  c = Choice::DEQ;
132  invoke(m->deq_q.pop());
133  }
134  DVLOG(2) << "combining: tail(" << m->tail << "), tail_allocator(" << m->tail_allocator << ")";
135  m->ce.wait(SuspendedDelegate::create([self,m,c] {
136  switch (c) {
137  case Choice::PUSH: m->tail = m->tail_allocator; break;
138  case Choice::POP: m->tail_allocator = m->tail; break;
139  case Choice::DEQ: m->head = m->head_allocator; break;
140  }
141  if (m->has_requests()) {
142  master_combine(self);
143  } else {
144  m->combining = false;
145  }
146  DVLOG(2) << "woken: tail(" << m->tail << "), tail_allocator(" << m->tail_allocator << ")";
147  }));
148  }
149  return;
150  } // else: do second level of combining
151 
152  DVLOG(2) << "combining: tail(" << m->tail << "), tail_allocator(" << m->tail_allocator << ")";
153  long ncombined = 0;
154 
155  while ( !(m->push_q.blocked || m->push_q.empty()) ) {
156  ++ncombined;
157  m->ce.enroll();
158  invoke(m->push_q.pop());
159  }
160  while ( !(m->deq_q.blocked || m->deq_q.empty()) ) {
161  ++ncombined;
162  m->ce.enroll();
163  invoke(m->deq_q.pop());
164  }
165  m->ce.wait(SuspendedDelegate::create([self,m,ncombined] {
166  DVLOG(2) << "after pushes: tail(" << m->tail << "), tail_allocator(" << m->tail_allocator << ")";
167  m->tail = m->tail_allocator;
168  m->head = m->head_allocator;
169  long ncombined2 = ncombined;
170  while (!m->pop_q.empty()) {
171  ++ncombined2;
172  m->ce.enroll();
173  invoke(m->pop_q.pop());
174  }
175  global_vector_master_combined += ncombined2;
176  m->ce.wait(SuspendedDelegate::create([self,m] {
177  DVLOG(2) << "after pops: tail(" << m->tail << "), tail_allocator(" << m->tail_allocator << ")";
178  m->tail_allocator = m->tail;
179  // find new combiner...
180  if (m->has_requests()) {
181  master_combine(self);
182  } else {
183  m->combining = false;
184  }
185  }));
186  }));
187  }

191  using R = decltype(func());

193  FullEmpty<R> result;

199  auto val = func();

201 
202  if (result_addr.core() == mycore()) set_result();
203  else send_heap_message(result_addr.core(), set_result);
204  }));

207  if (MASTER == mycore()) do_call(); else send_message(MASTER, do_call);

267  inline void incr_with_wrap(size_t * i, long incr) {
268  long val = *i;
269  val += incr;
270  if (val >= capacity) {
271  val %= capacity;
272  } else if (val < 0){
273  val += capacity;
274  }
275  DCHECK_GE(val,0);
276  *i = val;
277  }

279  template< typename Cache >
280  void cache_with_wraparound(size_t start, size_t nelem, T * buffer) {
281  CHECK_GE(start, 0); CHECK_LT(start, capacity); CHECK_GE(start+nelem, 0);
282  struct Range {size_t start, end; };
283  if (start+nelem <= capacity) {
284  DVLOG(3) << "put(base[" << start << "]<" << (base+start).core() << ":" << (base+start).pointer() << ">, buffer:" << buffer << ", nelem:" << nelem << ")\n" << base;
285  Cache c(base+start, nelem, buffer); c.block_until_acquired();
286  } else {
287  auto end = (start+nelem)%capacity;
288  Cache c1(base, end, buffer);
289  Cache c2(base+start, capacity-start, buffer+end);
290  c1.start_acquire();
291  c2.block_until_acquired();
292  c1.block_until_acquired();
293  }

378  void push(const T& e) {
379  ++global_vector_push_ops;
380  double t = Grappa::walltime();
381  if (FLAGS_flat_combining) {
382  proxy.combine([&e](Proxy& p) {
383  if (p.npop > 0) {
384  ++global_vector_matched_pushes;
385  *p.pops[--p.npop] = e;
386  return FCStatus::MATCHED;
387  } else {
388  p.buffer[p.npush++] = e;

392  } else {
393  T val = e;
394  ++global_vector_push_msgs;

396  }
397  global_vector_push_latency += (Grappa::walltime() - t);
