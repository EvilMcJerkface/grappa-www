236  double t = Grappa::walltime();
237 
238  Grappa::call_on_all_cores([]{
239  Grappa::impl::global_scheduler.allow_active_workers(FLAGS_io_blocks_per_node);
240  });
241 
242  const size_t NBUF = FLAGS_io_blocksize_mb*(1L<<20)/sizeof(T);
243 
244  Grappa::CompletionEvent io_joiner;
245  auto joiner_addr = make_global(&io_joiner);
246 
247  read_array_args<T> args(f.fname, array, 0, nelem, joiner_addr);
248  args.file_offset = f.offset;
249  auto arg_addr = make_global(&args);

253  CHECK( i+n <= nelem) << "nelem = " << nelem << ", i+n = " << i+n;

264  });

338  // make directory with mode 777

346  }

351 
352  char dir[256]; strcpy(dir, dirname);
353  auto g_dir = make_global(dir);
354 
355  Grappa::on_all_cores([g_dir,namelen,array,nelems]{
356  range_t r = blockDist(0, nelems, Grappa::mycore(), Grappa::cores());
357 
358  char dir[256];
359  Incoherent<char>::RO c(g_dir,namelen+1,dir);
360 
361  char fname[FNAME_LENGTH]; array_dir_fname(fname, &c[0], r.start, r.end);
362  std::fstream fo(fname, std::ios::out | std::ios::binary);
363 
364  VLOG(1) << "saving to " << fname;
365 
366  const size_t NBUF = FLAGS_io_blocksize_mb*(1L<<20)/sizeof(T);
367  T * buf = Grappa::locale_alloc<T>(NBUF);
368 
369  for_buffered (i, n, r.start, r.end, NBUF) {
370  typename Incoherent<T>::RO c(array+i, n, buf);
371  c.block_until_acquired();
372  fo.write((char*)buf, sizeof(T)*n);
373  VLOG(1) << "wrote " << i << ".." << i+n << " (" << n << ")";
374  }
375  Grappa::locale_free(buf);
376 
377  fo.close();
378  VLOG(1) << "finished saving array[" << r.start << ":" << r.end << "]";
379  });
380 
381  t = Grappa::walltime() - t;
382  VLOG(2) << "save_array_time: " << t;
383  VLOG(2) << "save_rate_mbps: " << ((double)nelems * sizeof(T) / (1L<<20)) / t;
384  }
