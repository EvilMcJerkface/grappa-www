248  void _read_array_file(File& f, GlobalAddress<T> array, size_t nelem) {
249  double t = Grappa::walltime();
250 
251  Grappa::call_on_all_cores([]{
252  Grappa::impl::global_scheduler.allow_active_workers(FLAGS_io_blocks_per_node);
253  });
254 
255  const int64_t NBUF = FLAGS_io_blocksize_mb*(1L<<20)/sizeof(T);
256 
257  Grappa::CompletionEvent io_joiner;
258  auto joiner_addr = make_global(&io_joiner);
259 
260  read_array_args<T> args(f.fname, array, 0, nelem, joiner_addr);
261  args.file_offset = f.offset;
262  auto arg_addr = make_global(&args);
263 
264  // read array

266  CHECK( i+n <= nelem) << "nelem = " << nelem << ", i+n = " << i+n;
267 
268  io_joiner.enroll();
269  spawn<TaskMode::Unbound>([arg_addr, i, n]{
270  read_array_args<T>::task(arg_addr, i, n);

274 
275  Grappa::call_on_all_cores([]{
276  Grappa::impl::global_scheduler.allow_active_workers(-1);
277  });
278 
279  f.offset += nelem * sizeof(T);
280  t = Grappa::walltime() - t;
281  VLOG(1) << "read_array_time: " << t;
