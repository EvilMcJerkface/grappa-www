203 template <typename T>
204 bool GlobalQueue<T>::push( GlobalAddress<T> chunk_base, uint64_t chunk_amount ) {
205  CHECK( initialized );
206  DVLOG(5) << "push() base:" << chunk_base << " amount:" << chunk_amount;
207 
208  GlobalAddress< QueueEntry<T> > loc = Grappa_delegate_func< bool, GlobalAddress< QueueEntry<T> >, GlobalQueue<T>::push_reserve_g > ( false, HOME_NODE );
209  size_t msg_bytes = Grappa_sizeof_delegate_func_request< bool, GlobalAddress< QueueEntry<T> > >( );
210 
211  DVLOG(5) << "push() reserve done -- loc:" << loc;
212 
213  if ( loc.pointer() == NULL ) {
214  Grappa::Metrics::global_queue_stats.record_push_reserve_request( msg_bytes, false );
215  // no space in global queue; push failed
216  return false;
217  }
218 
219  Grappa::Metrics::global_queue_stats.record_push_reserve_request( msg_bytes, true );
220 
221  // push the queue entry that points to my chunk
222  ChunkInfo<T> c;
223  c.base = chunk_base;
224  c.amount = chunk_amount;
225  push_entry_args<T> entry_args;
226  entry_args.target = loc;
227  entry_args.chunk = c;
228  DVLOG(5) << "push() sending entry to " << loc;
229  bool had_sleeper = Grappa_delegate_func< push_entry_args<T>, bool, GlobalQueue<T>::push_entry_g > ( entry_args, loc.core() );
230  size_t entry_msg_bytes = Grappa_sizeof_delegate_func_request< push_entry_args<T>, bool >( );
231  Grappa::Metrics::global_queue_stats.record_push_entry_request( entry_msg_bytes, had_sleeper );
232 
233  return true;
234 }

242 template <typename T>
243 void GlobalQueue<T>::pull( ChunkInfo<T> * result ) {
244  CHECK( initialized );
245  DVLOG(5) << "pull";
246 
247  // blocking request for an address of an entry in the global queue
248  A_Entry loc;
249  D_A_Entry qdesc( &loc );
250  A_D_A_Entry desc_addr = make_global( &qdesc );
251  Grappa_call_on( HOME_NODE, GlobalQueue<T>::pull_reserve_am_g, &desc_addr ); // FIXME: call_on deprecated
252  size_t resv_msg_bytes = Grappa_sizeof_message( &desc_addr );
253  /* wait for element: this is designed to block forever if there are no more items shared in this queue
254  * for the rest of the program. */
255  qdesc.wait();
256 
257  CHECK( loc.pointer() != NULL ) << "Invalid global address. Pull is always blocking";
258 
259  Grappa::Metrics::global_queue_stats.record_pull_reserve_request( resv_msg_bytes );
260 
261  // get the element of the queue, which will point to data
262  Descriptor< ChunkInfo<T> > cdesc( result );
263 
264  pull_entry_args<T> entry_args;
265  entry_args.target = loc;
266  entry_args.descriptor = make_global( &cdesc );
267  Grappa_call_on( loc.core(), pull_entry_request_g_am, &entry_args ); // FIXME: call_on deprecated
268 
269  size_t entry_msg_bytes = Grappa_sizeof_message( &entry_args );
270  Grappa::Metrics::global_queue_stats.record_pull_entry_request( entry_msg_bytes );
271  cdesc.wait();
272 }

284 template <typename T>
285 A_Entry GlobalQueue<T>::push_reserve ( bool ignore ) {
286  CHECK( isMaster() );
287 
288  Grappa::Metrics::global_queue_stats.record_push_reserve_reply( Grappa_sizeof_delegate_func_reply< bool, A_Entry >() );
289 
290  DVLOG(5) << "push_reserve";
291 
292  CHECK( capacity > 0 );
293  if ( (tail % capacity == head % capacity) && (tail != head) ) {
294  return make_global( static_cast< QueueEntry<T> * >( NULL ) ); // no room
295  } else {
296  A_Entry assigned = queueBase + (tail % capacity);
297  tail++;
298 
299  // if there are any consumers, wake oldest and give the address just produced
300  if ( pullReserveWaiters.size() > 0 ) {
301  CHECK( head == tail-1 ) << "Size should be exactly one, since there are waiters and one value was just produced";
302  DVLOG(5) << "push_reserve: found waiters";
303 
304  A_Entry granted = assigned;
305  head++;
306 
307  A_D_A_Entry w = pullReserveWaiters.front();
308  pullReserveWaiters.pop();
309 
310  pull_reserve_sendreply( w, &granted, true );
311  }
312 
313  return assigned;
314  }

317 template <typename T>
318 void GlobalQueue<T>::pull_reserve ( A_D_A_Entry requestor ) {
319  CHECK( isMaster() );
320  DVLOG(5) << "pull_reserve";
321 
322  if ( tail == head ) {
323  // TODO: fixme, added cast that is wrong
324  pullReserveWaiters.push( requestor );
325  DVLOG(5) << "empty, must sleep. waiters queue now has " << pullReserveWaiters.size() << " elements";
326  } else {
327  A_Entry granted = queueBase + (head % capacity);
328  head++;
329 
330  pull_reserve_sendreply( requestor, &granted, false );
331  }

347 template <typename T>
348 bool GlobalQueue<T>::push_entry ( push_entry_args<T> args ) {
349  QueueEntry<T> * e = args.target.pointer();
350 
351  e->chunk = args.chunk;
352  e->valid = true;
353 
354  // if a consumer is waiting then send a wake message
355  if ( e->sleeper.pointer() != NULL ) {
356  DVLOG(5) << "push_entry: was sleeping consumer " << e->sleeper;
357 
358  pull_entry_sendreply( e->sleeper, e );
359  e->sleeper = make_global( static_cast<Descriptor< ChunkInfo<T> > * >( NULL ) );
360  return true;
361  } else {
362  DVLOG(5) << "push_entry: no consumer";
363  return false;
364  }
365 }

367 template <typename T>
368 void GlobalQueue<T>::pull_entry_request( pull_entry_args<T> * args ) {
369  // for now just a valid bit: empty or full, no nack for retry
370 
371  QueueEntry<T> * e = args->target.pointer();
372  if ( !e->valid ) {
373  // leave a note to wake the client.
374  // send no reply yet; the client will block
375  e->sleeper = args->descriptor;
376  } else {
377  pull_entry_sendreply( args->descriptor, e );
378  }
379 }
