80 template <typename T, typename AllReducerType, typename CF>
81 T reduction(T init, CF f) {
82  Core master = Grappa::mycore();
83  auto r = Grappa::symmetric_global_alloc<AllReducerType>();
84  Grappa::on_all_cores( [=] {
85  // call the constructor
86  new (r.localize()) AllReducerType(init);
87  r->reset();
88  });
89 
90  // user code, calling accumulate
91  f(r);
92 
93  T result;
94  Grappa::on_all_cores( [master,r,&result] {
95  T localcopy = r->finish();
96  if (Grappa::mycore() == master)
97  result = localcopy;
98  });
99 
100  // FIXME: memory leak on r; but free is problematic with symmetric_global_alloc
101  // Grappa::global_free(r);
102 
103  return result;
104 }
