160 static inline void * coro_invoke(Worker * me, Worker * to, void * val) {
161 #ifdef CORO_PROTECT_UNUSED_STACK
162  if( to->base != NULL ) {
163  checked_mprotect( (void*)((intptr_t)to->base + 4096), to->ssize, PROT_READ | PROT_WRITE );
164  checked_mprotect((void*)(to), 4096, PROT_READ | PROT_WRITE );
165  }
166 
167  // compute expected stack pointer
168 
169  // get current stack pointer
170  void * rsp = NULL;
171  asm volatile("mov %%rsp, %0" : "=r"(rsp) );
172 
173  // adjust by register count
174  // TODO: couple this with save/restore strategies in stack.S
175  int num_registers_to_save = 8;
176  intptr_t stack_with_regs = (intptr_t)rsp - 8*1 - 8*num_registers_to_save; // 8-byte PC + 8-byte saved registers
177  me->guess = (void*) stack_with_regs; // store for debugging
178 
179  // round down to 4KB pages
180  int64_t size = (stack_with_regs - ((intptr_t)me->base + 4096)) & 0xfffffffffffff000L;
181  if( me != to && // don't protect if we're switching to ourselves
182  me->base != NULL && // don't protect if it's the native host thread
183  size > 0 ) {
184  checked_mprotect( (void*)((intptr_t)me->base + 4096), size, PROT_READ );
185  checked_mprotect( (void*)(me), 4096, PROT_READ );
186  }
187 #endif
188 
189  me->running = 0;
190  to->running = 1;
191 
192  val = swapstacks_inline(&(me->stack), &(to->stack), val);
193  return val;
194 }

203 static void tramp(Worker * me, void * arg) {
204  // Pass control back and forth a few times to get the info we need.
205  Worker * master = (Worker *) arg;
206  Worker * my_thr = (Worker *) coro_invoke(me, master, NULL);
207  thread_func f = (thread_func) coro_invoke(me, master, NULL);
208  void * f_arg = coro_invoke(me, master, NULL);
209  // Next time we're invoked, it'll be for real.
210  coro_invoke(me, master, NULL);
211 
212  StateTimer::setThreadState( StateTimer::SYSTEM );
213  StateTimer::enterState_system();
214 
215  // create new Tau task, and top level timer for the task
216 #ifdef GRAPPA_TRACE
217  int new_taskid;
218  TAU_CREATE_TASK(new_taskid);
219  my_thr->tau_taskid = new_taskid;
220  thread_last_tau_taskid = new_taskid;
221  GRAPPA_PROFILE_CREATE( mainprof, "start_thread", "()", TAU_DEFAULT );
222  GRAPPA_PROFILE_THREAD_START( mainprof, my_thr );
223 #endif
224 
225  // call thread target function
226  f(my_thr, f_arg);
227 
228  // stop top level Tau task timer
229  GRAPPA_PROFILE_THREAD_STOP( mainprof, my_thr );
230 
231  // We shouldn't return, but if we do, kill the Worker.
232  thread_exit(my_thr, NULL);
233 }
