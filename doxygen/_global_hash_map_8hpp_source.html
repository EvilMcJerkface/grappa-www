66  std::pair<bool,V> lookup(K key) {
67 
68  for (auto& e : this->entries) if (e.key == key) return std::pair<bool,K>{true, e.val};
69  return std::pair<bool,K>(false,V());
70  }
71  void insert(const K& key, const V& val) {
72  bool found = false;
73  for (auto& e : entries) {
74  if (e.key == key) {
75  e.val = val;
76  found = true;
77  break;
78  }
79  }
80  if (!found) {
81  entries.emplace_back(key, val);
82  }
83  }

115  void sync() {
116  CompletionEvent ce(map.size()+lookups.size());
117  auto cea = make_global(&ce);
118 
119  for (auto& e : map) { auto& k = e.first; auto& v = e.second;
120  ++hashmap_insert_msgs;
121  auto cell = owner->base+owner->computeIndex(k);
122  send_heap_message(cell.core(), [cell,cea,k,v]{
123  cell->insert(k, v);
124  complete(cea);
125  });
126  }
127 
128  for (auto& e : lookups) { auto k = e.first;
129  ++hashmap_lookup_msgs;
130  auto re = e.second;
131  DVLOG(3) << "lookup " << k << " with re = " << re;
132  auto cell = owner->base+owner->computeIndex(k);
133 
134  send_heap_message(cell.core(), [cell,k,cea,re]{
135  Cell * c = cell.localize();
136  bool found = false;
137  V val;
138  for (auto& e : c->entries) if (e.key == k) {
139  found = true;
140  val = e.val;
141  break;
142  }
143  send_heap_message(cea.core(), [cea,re,found,val]{
144  ResultEntry * r = re;
145  while (r != nullptr) {
146  r->found = found;
147  r->val = val;
148  r = r->next;
149  }
150  complete(cea);
151  });
152  });
153  }
154  ce.wait();
155  }
156  };

262 template< SyncMode S = SyncMode::Blocking,
263  GlobalCompletionEvent * C = &impl::local_gce,
264  typename K = nullptr_t, typename V = nullptr_t,
265  typename F = nullptr_t >
266 void insert(GlobalAddress<GlobalHashMap<K,V>> self, K key, F on_insert) {
267  ++hashmap_insert_msgs;
268  delegate::call<S,C>(self->base+self->computeIndex(key),
269  [=](typename GlobalHashMap<K,V>::Cell& c){
270  for (auto& e : c.entries) {
271  if (e.key == key) {
272  on_insert(e.val);
273  return;
274  }
275  }
276  c.entries.emplace_back(key);
277  on_insert(c.entries.back().val);
278  });
279 }
