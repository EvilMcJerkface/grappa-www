141  template< typename M, typename F >
142  inline auto call(Core dest, M mutex, F func) -> decltype(func(mutex())) {
143  using R = decltype(func(mutex()));
144  delegate_ops++;
145 
146  if (dest == mycore()) {
147  delegate_targets++;
148  delegate_short_circuits++;
149  // auto l = mutex();
150  // lock(l);
151  auto r = func(mutex());
152  // unlock(l);
153  return r;
154  } else {
155  FullEmpty<R> result;
156  auto result_addr = make_global(&result);
157  auto set_result = [result_addr](const R& val){
158  send_heap_message(result_addr.core(), [result_addr,val]{
159  result_addr->writeXF(val);
160  });
161  };
162 
163  send_message(dest, [set_result,mutex,func] {
164  delegate_targets++;
165  auto l = mutex();
166  if (is_unlocked(l)) { // if lock is not held
167  // lock(l);
168  set_result(func(l));
169  } else {
170  add_waiter(l, SuspendedDelegate::create([set_result,func,l]{
171  // lock(l);
172  CHECK(is_unlocked(l));
173  set_result(func(l));
174  }));
175  }
176  });
177  auto r = result.readFE();
178  return r;
179  }
180  }

189  template <typename F>
190  inline auto call_suspendable(Core dest, F func) -> decltype(func()) {
191  delegate_ops++;
192  using R = decltype(func());
193  Core origin = Grappa::mycore();
194 
195  if (dest == origin) {
196  delegate_targets++;
197  delegate_short_circuits++;
198  return func();
199  } else {
200  FullEmpty<R> result;
201  int64_t network_time = 0;
202  int64_t start_time = Grappa::timestamp();
203 
204  send_message(dest, [&result, origin, func, &network_time, start_time] {
205  delegate_targets++;
206 
207  spawn([&result, origin, func, &network_time, start_time] {
208  R val = func();
209  // TODO: replace with handler-safe send_message
210  send_heap_message(origin, [&result, val, &network_time, start_time] {
211  network_time = Grappa::timestamp();
212  record_network_latency(start_time);
213  result.writeXF(val); // can't block in message, assumption is that result is already empty
214  });
215  });
216  }); // send message
217  // ... and wait for the result
218  R r = result.readFE();
219  record_wakeup_latency(start_time, network_time);
220  return r;
221  }

283  template < typename T, typename U >
284  class FetchAddCombiner {
285  // TODO: generalize to define other types of combiners
286  private:
287  // configuration
288  const GlobalAddress<T> target;
289  const U initVal;
290  const uint64_t flush_threshold;
291 
292  // state
293  T result;
294  U increment;
295  uint64_t committed;
296  uint64_t participant_count;
297  uint64_t ready_waiters;
298  bool outstanding;
299  ConditionVariable untilNotOutstanding;
300  ConditionVariable untilReceived;
301 
302  // wait until fetch add unit is in aggregate mode
303  // TODO: add concurrency (multiple fetch add units)
304  void block_until_ready() {
305  while ( outstanding ) {
306  ready_waiters++;
307  Grappa::wait(&untilNotOutstanding);
308  ready_waiters--;
309  }
310  }
311 
312  void set_ready() {
313  outstanding = false;
314  Grappa::broadcast(&untilNotOutstanding);
315  }
316 
317  void set_not_ready() {
318  outstanding = true;
319  }
320 
321  public:
322  FetchAddCombiner( GlobalAddress<T> target, uint64_t flush_threshold, U initVal )
323  : target( target )
324  , initVal( initVal )
325  , flush_threshold( flush_threshold )
326  , result()
327  , increment( initVal )
328  , committed( 0 )
329  , participant_count( 0 )
330  , ready_waiters( 0 )
331  , outstanding( false )
332  , untilNotOutstanding()
333  , untilReceived()
334  {}

347  void promise() {
348  committed += 1;
349  }
350  // because tasks run serially, promise() replaces the flat combining tree
351 
352  T fetch_and_add( U inc ) {
353 
354  block_until_ready();
355 
356  // fetch add unit is now aggregating so add my inc
357 
358  participant_count++;
359  committed--;
360  increment += inc;
361 
362  // if I'm the last entered client and either the flush threshold
363  // is reached or there are no more committed participants then start the flush
364  if ( ready_waiters == 0 && (participant_count >= flush_threshold || committed == 0 )) {
365  set_not_ready();
366  uint64_t increment_total = increment;
367  flat_combiner_fetch_and_add_amount += increment_total;
368  auto t = target;
369  result = call(target.core(), [t, increment_total]() -> U {
370  T * p = t.pointer();
371  uint64_t r = *p;
372  *p += increment_total;
373  return r;
374  });
375  // tell the others that the result has arrived
376  Grappa::broadcast(&untilReceived);
377  } else {
378  // someone else will start the flush
379  Grappa::wait(&untilReceived);
380  }
381 
382  uint64_t my_start = result;
383  result += inc;
384  participant_count--;
385  increment -= inc; // for validation purposes (could just set to 0)
386  if ( participant_count == 0 ) {
387  CHECK( increment == 0 ) << "increment = " << increment << " even though all participants are done";
388  set_ready();
389  }
390 
391  return my_start;
392  }
393  };
