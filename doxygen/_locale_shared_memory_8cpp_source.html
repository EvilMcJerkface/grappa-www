65 void LocaleSharedMemory::create() {
66  region_size = FLAGS_locale_shared_size;
67  VLOG(2) << "Creating LocaleSharedMemory region " << region_name
68  << " with " << region_size << " bytes"
69  << " on " << global_communicator.mycore()
70  << " of " << global_communicator.cores();
71 
72  // if possible, delete this user's old leftover share memory regions
73  // if this returns false, either there was nothing to delete or
74  // there was something owned by somebody else.
75  bool deleted_one = boost::interprocess::shared_memory_object::remove( region_name.c_str() );
76  if( deleted_one ) {
77  VLOG(1) << "Deleted an old, leftover " << region_name.c_str();
78  }
79 
80  // now, try to allocate a new one
81  try {
82  segment = boost::interprocess::fixed_managed_shared_memory( boost::interprocess::create_only,
83  region_name.c_str(),
84  region_size,
85  base_address);
86  }
87  catch(...){
88  LOG(ERROR) << "Failed to create locale shared memory of size " << region_size;
89  boost::interprocess::shared_memory_object::remove( region_name.c_str() );
90  failure_function();
91  throw;
92  }
93  VLOG(2) << "Created LocaleSharedMemory region " << region_name
94  << " with " << region_size << " bytes"
95  << " on " << global_communicator.mycore()
96  << " of " << global_communicator.cores();
97 }

99 void LocaleSharedMemory::attach() {
100  // first, check that we can create something large enough
101  CHECK_GE( FLAGS_locale_shared_size, global_bytes_per_locale )
102  << "Can't accomodate " << global_bytes_per_locale << " bytes of shared heap per locale with "
103  << FLAGS_locale_shared_size << " total shared bytes.";
104 
105 
106  VLOG(2) << "Attaching to LocaleSharedMemory region " << region_name
107  << " on " << global_communicator.mycore()
108  << " of " << global_communicator.cores();
109  try {
110  segment = boost::interprocess::fixed_managed_shared_memory( boost::interprocess::open_only,
111  region_name.c_str(),

113  }
114  catch(...){
115  LOG(ERROR) << "Failed to attach to locale shared memory of size " << region_size;
116  boost::interprocess::shared_memory_object::remove( region_name.c_str() );
117  failure_function();

120  VLOG(2) << "Attached to LocaleSharedMemory region " << region_name
121  << " on " << global_communicator.mycore()
