144  static const int64_t tick_scale = 1L; //(1L << 30);

146  Worker * nextCoroutine ( bool isBlocking=true ) {
147  scheduler_context_switches++;
148 
149  Grappa::Timestamp current_ts = 0;
150 #ifdef VTRACE_FULL
151  VT_TRACER("nextCoroutine");
152 #endif
153  CHECK_EQ( in_no_switch_region_, false ) << "Trying to context switch in no-switch region";
154 
155  do {
156  Worker * result;
157  scheduler_count++;
158 
159  // tick the timestap counter
160  Grappa::tick();
161  current_ts = Grappa::timestamp();
162 
163  // maybe sample
164  if( Grappa::impl::take_tracing_sample ) {
165  Grappa::impl::take_tracing_sample = false;
166 #ifdef HISTOGRAM_SAMPLED
167  DVLOG(3) << "sampling histogram";
168  Grappa::Metrics::histogram_sample();
169 #else
170  Grappa::Metrics::sample();
171 #endif
172  }
173 
174  if( ( global_communicator.mycore() == 0 ) &&
175  ( current_ts - prev_stats_blob_ts > FLAGS_stats_blob_ticks) ) {
176  prev_stats_blob_ts = current_ts;
177 
178  std::ofstream f(FLAGS_stats_blob_filename);
179  Grappa::Metrics::print(f);
180  f.close();
181  }
182 
183  // check for periodic tasks
184  result = periodicDequeue(current_ts);
185  if (result != NULL) {
186  // DVLOG(5) << current_thread->id << " scheduler: pick periodic";
187  *(stats.state_timers[ stats.prev_state ]) += (current_ts - prev_ts) / tick_scale;
188  stats.prev_state = TaskingSchedulerMetrics::StatePoll;
189  prev_ts = current_ts;
190  return result;
191  }
192 
193 
194  // check ready tasks
195  result = readyQ.dequeue();
196  if (result != NULL) {
197  // DVLOG(5) << current_thread->id << " scheduler: pick ready";
198  *(stats.state_timers[ stats.prev_state ]) += (current_ts - prev_ts) / tick_scale;
199  stats.prev_state = TaskingSchedulerMetrics::StateReady;
200  prev_ts = current_ts;
201  return result;
202  }
203 
204  // check if scheduler is allowed to have more active workers
205  if (num_active_tasks < max_allowed_active_workers) {
206  // check for new workers
207  result = getWorker();
208  if (result != NULL) {
209  // DVLOG(5) << current_thread->id << " scheduler: pick task worker";
210  *(stats.state_timers[ stats.prev_state ]) += (current_ts - prev_ts) / tick_scale;
211  stats.prev_state = TaskingSchedulerMetrics::StateReady;
212  prev_ts = current_ts;
213  return result;
214  }
215  }
216 
217 
218  if (FLAGS_poll_on_idle) {
219  *(stats.state_timers[ stats.prev_state ]) += (current_ts - prev_ts) / tick_scale;
220 
221  if( FLAGS_rdma_flush_on_idle ) {
222  Grappa::impl::idle_flush_rdma_aggregator();
223  }
224 
225  if ( idle_flush_aggregator() ) {
226  stats.prev_state = TaskingSchedulerMetrics::StateIdleUseful;
227  } else {
228  stats.prev_state = TaskingSchedulerMetrics::StateIdle;
229  }
230 
231 
232  StateTimer::enterState_scheduler();
233  } else {
234  *(stats.state_timers[ stats.prev_state ]) += (current_ts - prev_ts) / tick_scale;
235  stats.prev_state = TaskingSchedulerMetrics::StateIdle;
236  usleep(1);
237  }
238 
239  prev_ts = current_ts;
240  // no coroutines can run, so handle
241  /*DVLOG(5) << current_thread->id << " scheduler: no coroutines can run"
242  << "[isBlocking=" << isBlocking
243  << " periodQ=" << (periodicQ.empty() ? "empty" : "full")
244  << " unassignedQ=" << (unassignedQ.empty() ? "empty" : "full") << "]";*/
245  // usleep(1);
246  } while ( isBlocking || !queuesFinished() );
247  // exit if all threads exited, including idle workers
248  // TODO just as use mightBeWork as shortcut, also kill all idle unassigned workers on cbarrier_exit
249  return NULL;
250  }

429 inline bool TaskingScheduler::thread_yield( ) {
430  CHECK( current_thread != master ) << "can't yield on a system Worker";
431  StateTimer::enterState_scheduler();
432 
433  ready( current_thread );
434 
435  Worker * yieldedThr = current_thread;
436 
437  Worker * next = nextCoroutine( );
438  bool gotRescheduled = (next == yieldedThr);
439 
440  current_thread = next;
441  //DVLOG(5) << "Worker " << yieldedThr->id << " yielding to " << next->id << (gotRescheduled ? " (same thread)." : " (diff thread).");
442  thread_context_switch( yieldedThr, next, NULL);
443 
444  return gotRescheduled; // 0=another ran; 1=me got rescheduled immediately
445 }

541 inline bool TaskingScheduler::thread_idle( uint64_t total_idle ) {
542  CHECK( num_idle+1 <= total_idle ) << "number of idle threads should not be more than total"
543  << " (" << num_idle+1 << " / " << total_idle << ")";
544  if (num_idle+1 == total_idle) {
545  DVLOG(5) << "going idle and is last";
546  } else {
547  DVLOG(5) << "going idle and is " << num_idle+1 << " of " << total_idle;
548  }
549 
550  num_idle++;
551  current_thread->idle = 1;
552 
553  unassigned( current_thread );
554 
555  thread_suspend( );
556 
557  // woke so decrement idle counter
558  num_idle--;
559  current_thread->idle = 0;
560 
561  return true;
562 }
