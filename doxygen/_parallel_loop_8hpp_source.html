80  void loop_decomposition(int64_t start, int64_t iterations, F loop_body) {
81  DVLOG(4) << "< " << start << " : " << iterations << ">";
82 
83  if (iterations == 0) {
84  return;
85  } else if ((Threshold == USE_LOOP_THRESHOLD_FLAG && iterations <= FLAGS_loop_threshold)
86  || iterations <= Threshold) {
87  loop_body(start, iterations);
88  return;
89  } else {
90  // spawn right half
91  int64_t rstart = start+(iterations+1)/2, riters = iterations/2;
92  Core origin = mycore();
93 
94  // pack these 3 into 14 bytes so we still have room for a full 8-byte word from user
95  // (also need to count 2 bytes from lambda overhead or something)
96  struct { long rstart:48, riters:48, origin:16; } packed = { rstart, riters, origin };
97 
98  if (C) C->enroll();
99  spawn<B>([packed, loop_body] {
100  loop_decomposition<B,C,Threshold>(packed.rstart, packed.riters, loop_body);
101  if (C) C->send_completion(packed.origin);
102  });
103 
104  // left side here
105  loop_decomposition<B,C,Threshold>(start, (iterations+1)/2, loop_body);
106  }
107  }

130  if (refs == 0) delete this;

327  template< typename T >
328  std::pair<Core,Core> cores_with_elements(GlobalAddress<T> base, size_t nelem) {
329  Core start_core = base.core();
330  Core nc = cores();
331  Core fc;
332  int64_t nbytes = nelem*sizeof(T);
333  auto end = base+nelem;
334  if (nelem > 0) { fc = 1; }
335 
336  size_t block_elems = block_size / sizeof(T);
337  int64_t nfirstcore = base.block_max() - base;
338  int64_t n = nelem - nfirstcore;
339 
340  if (n > 0) {
341  int64_t nrest = n / block_elems;
342  if (nrest >= nc-1) {
343  fc = nc;
344  } else {
345  fc += nrest;
346  if ((end - end.block_min()) && end.core() != base.core()) {
347  fc += 1;
348  }
349  }
350  }
351 
352  return std::make_pair( start_core, fc );
353  }
