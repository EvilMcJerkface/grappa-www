30 size_t round_up_page_size( size_t s ) {
31  // TODO: fix this for huge pages
32  const size_t page_size = 1 << 12;
33  size_t new_s = s;
34 
35  if( s < page_size ) {
36  new_s = page_size;
37  } else {
38  size_t diff = s % page_size;
39  if( 0 != diff ) {
40  new_s = s + page_size - diff;
41  }
42  }
43 
44  DVLOG(5) << "GlobalMemory rounding up from " << s << " to " << new_s;
45  return new_s;
46 }
