94 void Communicator::init( int * argc_p, char ** argv_p[] ) {
95 #ifdef HEAPCHECK_ENABLE
96  {
97  HeapLeakChecker::Disabler disable_leak_checks_here;
98 #endif
99  GASNET_CHECK( gasnet_init( argc_p, argv_p ) );
100 #ifdef HEAPCHECK_ENABLE
101  }
102 #endif
103  // make sure the Core type is big enough for our system
104  assert( static_cast< int64_t >( gasnet_nodes() ) <= (1L << sizeof(Core) * 8) &&
105  "Core type is too small for number of nodes in job" );
106 
107  // initialize job geometry
108  mycore_ = gasnet_mynode();
109  cores_ = gasnet_nodes();
110 
111  mylocale_ = gasneti_nodeinfo[mycore_].supernode;
112  locales_ = gasnet_nodes() / gasneti_nodemap_local_count;
113  locale_mycore_ = gasneti_nodemap_local_rank;
114  locale_cores_ = gasneti_nodemap_local_count;
115 
116  // allocate and initialize core-to-locale translation
117  locale_of_core_ = new Locale[ cores_ ];
118  for( int i = 0; i < cores_; ++i ) {
119  locale_of_core_[i] = gasneti_nodeinfo[i].supernode;
120  }
121 
122  DVLOG(2) << " mycore_ " << mycore_
123  << " cores_ " << cores_
124  << " mylocale_ " << mylocale_
125  << " locales_ " << locales_
126  << " locale_mycore_ " << locale_mycore_
127  << " locale_cores_ " << locale_cores_
128  << " pid " << getpid();
129 
130  registration_is_allowed_ = true;
131 }

138 void Communicator::activate() {
142  assert( registration_is_allowed_ );
143  GASNET_CHECK( gasnet_attach( &handlers_[0], handlers_.size(), // install active message handlers
144  SHARED_PROCESS_MEMORY_SIZE,

146  stats.reset_clock();
147  registration_is_allowed_ = false;
148  communication_is_allowed_ = true;

150  barrier();
151  DVLOG(3) << "Leaving activation barrier";
