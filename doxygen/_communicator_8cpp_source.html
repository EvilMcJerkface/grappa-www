252 size_t Communicator::adjust_footprint(size_t target) {
253 
254  auto& sends = FLAGS_log2_concurrent_sends;
255  auto& recvs = FLAGS_log2_concurrent_receives;
256  auto& size = FLAGS_log2_buffer_size;
257 
258  if (estimate_footprint() > target) {
259  if (mycore == 0) LOG(WARNING) << "Adjusting to fit in target footprint: " << target << " bytes";
260  while (estimate_footprint() > target) {
261  // first try adjusting number of concurrent buffers
262  if (sends > MIN_CONCURRENT_BUFFERS && sends >= recvs) sends--;
263  else if (recvs > MIN_CONCURRENT_BUFFERS && recvs >= sends) recvs--;
264  // next try making buffers smaller
265  else if (size > MIN_LOG2_BUFFER_SIZE) size--;
266  // otherwise we just have to return what we've got (we're at the minimum allowable footprint)
267  else break;
268  }
269  }
270 
271  if (mycore == 0) VLOG(2) << "\nAdjusted:"
272  << "\n estimated footprint: " << estimate_footprint()
273  << "\n log2_concurrent_sends: " << sends
274  << "\n log2_concurrent_receives: " << recvs
275  << "\n log2_buffer_size: " << size;
276  return estimate_footprint();
277 }

340 void Communicator::garbage_collect() {
341  int flag;
342  MPI_Status status;
343 
344  // check for completed sends and re-enable
345  while( send_tail != send_head ) {
346  auto c = &sends[send_tail];
347  if( c->reference_count > 0 ) {
348  MPI_CHECK( MPI_Test( &c->request, &flag, &status ) );
349  if( flag ) {
350  if( c->callback ) {
351  (c->callback)( c, status.MPI_SOURCE, status.MPI_TAG, c->size );
352  }
353  c->reference_count = 0;
354  send_tail = (send_tail + 1) & send_mask;
355  } else { // not sent yet
356  break;
357  }
358  } else { // not reference count > 0
359  break;
360  }
361  }
362 
363  // check external send contexts too
364  while( !external_sends.empty() ) {
365  auto c = external_sends.front();
366  if( c->reference_count > 0 ) {
367  MPI_CHECK( MPI_Test( &c->request, &flag, &status ) );
368  if( flag ) {
369  c->reference_count = 0;
370  if( c->callback ) {
371  (c->callback)( c, status.MPI_SOURCE, status.MPI_TAG, c->size );
372  }
373  external_sends.pop_front();
374  } else { // not sent yet
375  break;
376  }
377  } else { // not reference_count > 0
378  break;
379  }
380  }
381 }

383 void Communicator::repost_receive_buffers() {
384 
385  // consume buffers that are completely delivered
386  while( receive_tail != receive_dispatch ) {
387  auto c = &receives[receive_tail];
388  if( 0 == c->reference_count ) {
389  receive_tail = (receive_tail + 1) & receive_mask;
390  } else {
391  break;
392  }
393  }
394 
395  // repost free buffers
396  while( ((receive_head + 1) & receive_mask) != receive_tail ) {
397  auto c = &receives[receive_head];
398  post_receive( c );
399  receive_head = (receive_head + 1) & receive_mask;
400  }
401

416 void Communicator::process_received_buffers() {
417  MPI_Status status;
418 
419  while( receive_dispatch != receive_head ) {
420  int flag;
421  auto c = &receives[receive_dispatch];
422 
423  MPI_CHECK( MPI_Test( &c->request, &flag, &status ) );
424 
425  // if message has been received
426  if( flag ) {
427  int size = 0;
428  MPI_CHECK( MPI_Get_count( &status, MPI_BYTE, &size ) );
429  c->reference_count = 1;
430  // start delivering received buffer
431  receive( c, size );
432  if( c->callback ) {
433  (c->callback)( c, status.MPI_SOURCE, status.MPI_TAG, size );
434  }
435  receive_dispatch = (receive_dispatch + 1) & receive_mask;
436  // update if anything has finished delivery
437  repost_receive_buffers();
438  } else {
439  break;
440  }
441  }
442 
443  // see if anything else finished delivery while we were busy
444  repost_receive_buffers();
445 }
