70  void reset( ) {
71  CHECK( !release_started_ || released_ ) << "inconsistent state for reset";
72  release_started_ = false;
73  released_ = false;
74  thread_ = NULL;
75  num_messages_ = 0;
76  response_count_ = 0;
77  if( *count_ == 0 ) {
78  DVLOG(5) << "Zero-length release";
79  release_started_ = true;
80  released_ = true;
81  } else if( request_address_->is_2D() ) {
82  num_messages_ = 1;
83  if( request_address_->core() == Grappa::mycore() ) {
84  release_started_ = true;
85  released_ = true;
86  }
87  } else {
88  DVLOG(5) << "Straddle: block_max is " << (*request_address_ + *count_).block_max() ;
89  DVLOG(5) << ", request_address is " << *request_address_;
90  DVLOG(5) << ", sizeof(T) is " << sizeof(T);

94 
95  DVLOG(5) << "Straddle: address is " << *request_address_ ;
96  DVLOG(5) << ", address + count is " << *request_address_ + *count_;
97 
98  ptrdiff_t byte_diff = ( (*request_address_ + *count_ - 1).last_byte().block_max() -
99  request_address_->first_byte().block_min() );
100 
101  DVLOG(5) << "Straddle: address block max is " << request_address_->block_max();
102  DVLOG(5) << " address + count block max is " << (*request_address_ + *count_).block_max();

106  DVLOG(5) << " address block min " << request_address_->block_min();
107  DVLOG(5) << "Straddle: diff is " << byte_diff << " bs " << block_size;
108  num_messages_ = byte_diff / block_size;
109  }

112 
113  DVLOG(5) << "Detecting straddle for sizeof(T):" << sizeof(T)
114  << " count:" << *count_

116  << " request_address:" << *request_address_;

119  void start_release() {
120  if( !release_started_ ) {
121 #ifdef VTRACE_FULL
122  VT_TRACER("incoherent start_release");
123 #endif
124  DVLOG(5) << "Worker " << Grappa::current_worker()
125  << " issuing release for " << *request_address_
126  << " * " << *count_ ;
127  release_started_ = true;
128  size_t total_bytes = *count_ * sizeof(T);
129 
130  // allocate enough requests/messages that we don't run out
131  size_t nmsg = total_bytes / block_size + 2;
132  size_t msg_size = sizeof(Grappa::PayloadMessage<RequestArgs>);
133 
134  if (nmsg*msg_size < Grappa::current_worker()->stack_remaining()-8192) {
135  // try to put message storage on stack if there's space
136  char msg_buf[nmsg*msg_size];
137  Grappa::MessagePool pool(msg_buf, sizeof(msg_buf));
138  do_release(pool);
139  } else {
140  // fall back on heap-allocating the storage
141  Grappa::MessagePool pool(nmsg*msg_size);
142  do_release(pool);
143  }
144  }
145  }

147  void do_release(Grappa::impl::MessagePoolBase& pool) {
148  size_t total_bytes = *count_ * sizeof(T);
149 
150  RequestArgs args;
151  args.request_address = *request_address_;
152  DVLOG(5) << "Computing request_bytes from block_max " << request_address_->first_byte().block_max() << " and " << *request_address_;
153  args.reply_address = make_global( this );
154  size_t offset = 0;
155  size_t request_bytes = 0;
156 
157  for( size_t i = 0;
158  offset < total_bytes;
159  offset += request_bytes, i++) {
160 
161  request_bytes = args.request_address.first_byte().block_max() - args.request_address.first_byte();
162 
163  if( request_bytes > total_bytes - offset ) {
164  request_bytes = total_bytes - offset;
165  }
166 
167  DVLOG(5) << "sending release request with " << request_bytes
168  << " of total bytes = " << *count_ * sizeof(T)
169  << " to " << args.request_address;
170 
171  pool.send_message(args.request_address.core(),
172  [args](void * payload, size_t payload_size) {
173  IRMetrics::count_release_ams( payload_size );
174  DVLOG(5) << "Worker " << Grappa::current_worker()
175  << " received release request to " << args.request_address
176  << " reply to " << args.reply_address;
177  memcpy( args.request_address.pointer(), payload, payload_size );
178 
179  auto reply_address = args.reply_address;
180  Grappa::send_heap_message(args.reply_address.core(), [reply_address]{
181  DVLOG(5) << "Worker " << Grappa::current_worker() << " received release reply to " << reply_address;
182  reply_address.pointer()->release_reply();
183  });
184 
185  DVLOG(5) << "Worker " << Grappa::current_worker()
186  << " sent release reply to " << reply_address;
187  },
188  (char*)(*pointer_) + offset, request_bytes
189  );
190 
191  // TODO: change type so we don't screw with pointer like this
192  args.request_address = GlobalAddress<T>::Raw( args.request_address.raw_bits() + request_bytes );
193  }
194  DVLOG(5) << "release started for " << args.request_address;
195 
196  // blocks here waiting for messages to be sent
197  }

216  }

218  << " woke up for " << *request_address_

220  }
221  }

224  void release_reply( ) {
225  DVLOG(5) << "Worker " << Grappa::current_worker()
226  << " received release reply ";
227  ++response_count_;
228  if ( response_count_ == num_messages_ ) {
229  released_ = true;

232  << " waking Worker " << thread_;

234  }
