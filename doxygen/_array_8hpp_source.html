61  template< typename T >
62  void do_memcpy_locally(GlobalAddress<T> dst, GlobalAddress<T> src, size_t nelem) {
63  typedef typename Incoherent<T>::WO Writeback;
64  auto src_end = src+nelem;
65  // T * local_base = src.localize(), * local_end = (src+nelem).localize();
66  int64_t nfirstcore = src.block_max() - src;
67  int64_t nlastcore = src_end - src_end.block_min();
68  int64_t nmiddle = nelem - nfirstcore - nlastcore;
69 
70  const size_t nblock = block_size / sizeof(T);
71 
72  CHECK_EQ(nmiddle % nblock, 0);
73 
74  auto src_start = src;
75  if (src.core() == mycore()) {
76  int64_t nfirstcore = src.block_max() - src;
77  if (nfirstcore > 0 && nlastcore != nblock) {
78  DVLOG(3) << "nfirstcore = " << nfirstcore;
79  Writeback w(dst, nfirstcore, src.pointer());
80  src_start += nfirstcore;
81  }
82  }
83  if ((src_end-1).core() == mycore()) {
84  int64_t nlastcore = src_end - src_end.block_min();
85  int64_t index = nelem - nlastcore;
86  if (nlastcore > 0 && nlastcore != nblock) {
87  DVLOG(3) << "nlastcore = " << nlastcore << ", index = " << index;
88  CHECK((src+index).core() == mycore());
89  Writeback w(dst+index, nlastcore, (src+index).pointer());
90  src_end -= nlastcore;
91  }
92  }
93 
94  auto * local_base = src_start.localize();
95  size_t nlocal_trimmed = src_end.localize() - local_base;
96  CHECK_EQ((nlocal_trimmed) % nblock, 0);
97  size_t nlocalblocks = nlocal_trimmed/nblock;
98  Writeback * ws = locale_alloc<Writeback>(nlocalblocks);
99  for (size_t i=0; i<nlocalblocks; i++) {
100  size_t j = make_linear(local_base+(i*nblock))-src;
101  new (ws+i) Writeback(dst+j, nblock, local_base+(i*nblock));
102  ws[i].start_release();
103  }
104  for (size_t i=0; i<nlocalblocks; i++) { ws[i].block_until_released(); }
105  locale_free(ws);
106  }
107 }
