49  inline void record_wakeup_latency(int64_t start_time, int64_t network_time) {
50  auto current_time = Grappa::timestamp();
51  auto blocked_time = current_time - start_time;
52  auto wakeup_latency = current_time - network_time;
53  delegate_roundtrip_latency += blocked_time;
54  delegate_wakeup_latency += wakeup_latency;
55  }
