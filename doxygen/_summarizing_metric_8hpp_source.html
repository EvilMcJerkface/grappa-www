56  void process( T t ) {
57  n++;
58  double delta = t - mean;
59  mean += delta / n;
60  M2 += delta * (t - mean);
61  if (n == 1) {
62  min = t;
63  max = t;
64  }
65  if (t > max) max = t;
66  if (t < min) min = t;
67  }
68 
69  double variance() const {
70  return (n<2) ? 0.0 : M2 / (n - 1);
71  }
72 
73  double stddev() const {
74  return sqrt( variance() );
75  }
