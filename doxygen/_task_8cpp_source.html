111 void TaskManager::init ( Core localId_arg, Core * neighbors_arg, Core numLocalNodes_arg ) {
112  if ( FLAGS_load_balance.compare( "none" ) == 0 ) {
113  doSteal = false; doShare = false; doGQ = false;
114  } else if ( FLAGS_load_balance.compare( "steal" ) == 0 ) {
115  doSteal = true; doShare = false; doGQ = false;
116  } else if ( FLAGS_load_balance.compare( "share" ) == 0 ) {
117  CHECK( false ) << "--load_balance=share currently unsupported; see tasks/StealQueue.hpp";
118  doSteal = false; doShare = true; doGQ = false;
119  } else if ( FLAGS_load_balance.compare( "gq" ) == 0 ) {
120  CHECK( false ) << "--load_balance=gq currently unsupported; see tasks/StealQueue.hpp";
121  doSteal = false; doShare = false; doGQ = true;
122  } else {
123  CHECK( false ) << "load_balance=" << FLAGS_load_balance << "; must be {none, steal, share, gq}";
124  }
125 
126  fast_srand(0);
127 
128  localId = localId_arg;
129  neighbors = neighbors_arg;
130  numLocalNodes = numLocalNodes_arg;
131  chunkSize = FLAGS_chunk_size;
132 
133  // initialize neighbors to steal permutation
134  srandom(0);
135  for (int i=numLocalNodes; i>=2; i--) {
136  int ri = random() % i;
137  Core temp = neighbors[ri];
138  neighbors[ri] = neighbors[i-1];
139  neighbors[i-1] = temp;
140  }
141 }

278  if ( stealLock ) {
279 
280  GRAPPA_PROFILE_CREATE( prof, "stealing", "(session)", GRAPPA_TASK_GROUP );
281  GRAPPA_PROFILE_START( prof );
282 
283  // only one Worker is allowed to steal
284  stealLock = false;
285 
286  VLOG(5) << Grappa::current_worker() << " trying to steal";
287  int goodSteal = 0;
288  Core victimId = -1;
289 
290  for ( int64_t tryCount=0;
291  tryCount < numLocalNodes && !goodSteal && !(publicHasEle() || privateHasEle() || workDone);
292  tryCount++ ) {
293 
294  Core v = neighbors[nextVictimIndex];
295  victimId = v;
296  nextVictimIndex = (nextVictimIndex+1) % numLocalNodes;
297 
298  if ( v == Grappa::mycore() ) continue; // don't steal from myself
299 
300  goodSteal = publicQ.steal_locally(v, chunkSize);
