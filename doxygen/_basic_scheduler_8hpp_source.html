37  private:

45  // STUB: replace with real periodic threads
46  Grappa::Timestamp previous_periodic_ts;

48  Worker * periodicDequeue() {
49  // tick the timestap counter
50  Grappa::tick();
51  Grappa::Timestamp current_ts = Grappa::timestamp();
52 
53  if( current_ts - previous_periodic_ts > FLAGS_periodic_poll_ticks ) {
54  return periodicQ.dequeue();
55  } else {
56  return NULL;
57  }
58  }

61  Worker * nextCoroutine ( bool isBlocking=true ) {
62  do {
63  VLOG(5) << "scheduler: next coro";
64  Worker * result;
65 
66  // check for periodic tasks
67  result = periodicDequeue();
68  if (result != NULL) return result;
69 
70  // check ready tasks
71  result = readyQ.dequeue();
72  if (result != NULL) return result;
73 
74  // no coroutines can run, so handle
75  VLOG(5) << "scheduler: no coroutines can run";
76  usleep(1);
77  } while ( isBlocking || !periodicQ.empty() );
78 
79  VLOG(5) << "scheduler: returning coro NULL";
80  return NULL;
81  }

134 inline bool BasicScheduler::thread_yield( ) {
135  CHECK( current_thread != master ) << "can't yield on a system Worker";
136 
137  ready( current_thread );
138 
139  Worker * yieldedThr = current_thread;
140 
141  Worker * next = nextCoroutine( );
142  bool gotRescheduled = (next == yieldedThr);
143 
144  current_thread = next;
145  impl::thread_context_switch( yieldedThr, next, NULL);
146 
147  return gotRescheduled; // 0=another ran; 1=me got rescheduled immediately
148 }
