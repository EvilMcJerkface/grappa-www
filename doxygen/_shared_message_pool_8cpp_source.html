139  void* _shared_pool_alloc(size_t sz) {
140 // #ifdef DEBUG
141  // auto i = pool_stack->find(shared_pool);
142  // if (i >= 0) VLOG(0) << "found: " << shared_pool << ": " << i << " / " << pool_stack->size();
143 // #endif
144  DCHECK(!shared_pool || shared_pool->next == nullptr);
145  if (shared_pool && !shared_pool->emptying && shared_pool->remaining() >= sz) {
146  return _shared_pool_alloc_message(sz);
147  } else {
148  // if not emptying already, do it
149  auto *o = shared_pool;
150  shared_pool = nullptr;
151  if (o && !o->emptying) {
152  CHECK_GT(o->allocated, 0);
153  o->emptying = true;
154  if (o->to_send == 0) {
155  o->on_empty();
156  }
157  }
158  DCHECK(pool_stack->empty() || pool_stack->top->allocated == 0);
159  // find new shared_pool
160  SharedMessagePool *p = pool_stack->take();
161  if (p) {
162  CHECK_EQ(p->allocated, 0) << "not a fresh pool!";
163  shared_pool = p;
164  return _shared_pool_alloc_message(sz);
165  } else {
166  // can't allocate any more
167  // try to block until a pool frees up
168  do {
169  Grappa::wait(&blocked_senders);
170  if (shared_pool && !shared_pool->emptying && shared_pool->remaining() >= sz) {
171  p = shared_pool;
172  } else {
173  p = pool_stack->take();
174  }
175  } while (!p);
176 
177  // p must be able to allocate
178  shared_pool = p;
179  return _shared_pool_alloc_message(sz);
180  }
181  }
