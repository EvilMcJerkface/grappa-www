89 void activate() {
90  for( int i = 0; i < MAX_POOL_CACHELINE_COUNT; ++i ) {
91  auto message_size = (i+1) * CACHE_LINE_SIZE;
92  auto chunk_count = FLAGS_shared_pool_chunk_size / message_size;
93 
94  // only make big chunks for frequently-used message sizes
95  if( i >= (MAX_POOL_MESSAGE_SIZE_PREALLOCATED_CUTOFF / CACHE_LINE_SIZE) ) {
96  chunk_count = 1;
97  }
98 
99  aligned_pool_allocator_init( &message_pool[i],
100  CACHE_LINE_SIZE, // alignment
101  message_size,
102  chunk_count );
103  }
104 }

106 void* alloc(size_t sz) {
107  size_t cacheline_count = sz / CACHE_LINE_SIZE;
108  CHECK_EQ( cacheline_count * CACHE_LINE_SIZE, sz ) << "Message size not a multiple of cacheline size?";
109 
110  if( sz > MAX_POOL_MESSAGE_SIZE ) {
111  shared_pool_alloc_toobig++;
112  return locale_alloc_aligned<char>(CACHE_LINE_SIZE, sz);
113  } else {
114  // record the pool allocation (bucketed by number of cachelines)
115  switch( cacheline_count ) {
116  case 1: shared_pool_alloc_1cl++; break;
117  case 2: shared_pool_alloc_2cl++; break;
118  case 3: shared_pool_alloc_3cl++; break;
119  default: shared_pool_alloc_ncl++; break;
120  }
121  return aligned_pool_allocator_alloc( &message_pool[cacheline_count] );
122  }
123 }
