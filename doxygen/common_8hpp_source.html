61 inline double walltime(void) {
62 #if defined(__MTA__)
63  return((double)mta_get_clock(0) / mta_clock_freq());
64 #elif defined(__MACH__)
65  static mach_timebase_info_data_t info;
66  mach_timebase_info(&info);
67  uint64_t now = mach_absolute_time();
68  now *= info.numer;
69  now /= info.denom;
70  return 1.0e-9 * (double)now;
71 #else
72  struct timespec tp;
73 #if defined(CLOCK_PROCESS_CPUTIME_ID)
74 #define CLKID CLOCK_PROCESS_CPUTIME_ID
75 #elif defined(CLOCK_REALTIME_ID)
76 #define CLKID CLOCK_REALTIME_ID
77 #endif
78  clock_gettime(CLOCK_MONOTONIC, &tp);
79  return (double)tp.tv_sec + (double)tp.tv_nsec / BILLION;

126 inline unsigned int log2( unsigned int v ) {
127  unsigned int r; // result of log2(v) will go here
128  unsigned int shift;
129 
130  r = (v > 0xFFFF) << 4; v >>= r;
131  shift = (v > 0xFF ) << 3; v >>= shift; r |= shift;
132  shift = (v > 0xF ) << 2; v >>= shift; r |= shift;
133  shift = (v > 0x3 ) << 1; v >>= shift; r |= shift;
134  r |= (v >> 1);
135 
136  return r;
137 }

168 inline range_t blockDist(int64_t start, int64_t end, int64_t rank, int64_t numBlocks) {
169  int64_t numElems = end-start;
170  int64_t each = numElems / numBlocks;
171  int64_t remain = numElems % numBlocks;
172  int64_t mynum = (rank < remain) ? each+1 : each;
173  int64_t mystart = start + ((rank < remain) ? (each+1)*rank : (each+1)*remain + (rank-remain)*each);
174  range_t r = { mystart, mystart+mynum };
175  return r;
176 }

180 inline block_offset_t indexToBlock(int64_t index, int64_t numElements, int64_t numBlocks) {
181  block_offset_t result;
182  int64_t each = numElements / numBlocks,
183  remain = numElements % numBlocks;
184  if (index < (each+1)*remain) {
185  result = { index / (each+1), index % (each+1) };
186  } else {
187  index -= (each+1)*remain;
188  result = { remain + index/each, index % each };
189  }
190  // VLOG(1) << "result.block = " << result.block << ", remain = " << remain << ", index = " << index << ", each = " << each;
191  return result;
192 }
