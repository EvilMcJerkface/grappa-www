208  template< typename F >
209  void send_immediate( int dest, F f ) {
210  CommunicatorContext * c = NULL;
211  while( NULL == (c = try_get_send_context()) ) {
212  garbage_collect();
213  }
214 
215  DVLOG(3) << "Sending immediate " << &f << " to " << dest << " with " << c;
216  c->callback = NULL;
217  char * buf = (char*) c->buf;
218 
219  CHECK_LE( sizeof(Grappa::impl::Deserializer) + sizeof(f), c->size )
220  << "Immediate buffer size to small to contain "
221  << sizeof(Grappa::impl::Deserializer) << "-byte deserializer + "
222  << sizeof(f) << "-byte lambda";
223 
224  *((void**)buf) = (void*) Grappa::impl::immediate_deserializer<F>;
225  buf += sizeof(Grappa::impl::Deserializer);
226 
227  memcpy( buf, &f, sizeof(f) );
228  buf += sizeof(f);
229 
230  post_send( c, dest, (buf - ((char*) c->buf)) );
231  }
232 
233  template< typename F >
234  void send_immediate_with_payload( int dest, F f, void * payload, size_t payload_size ) {
235  CommunicatorContext * c = NULL;
236  while( NULL == (c = try_get_send_context()) ) {
237  garbage_collect();
238  }
239 
240  DVLOG(3) << "Sending immediate " << &f << " to " << dest << " with " << c;
241  c->callback = NULL;
242  char * buf = (char*) c->buf;
243 
244  CHECK_LE( sizeof(Grappa::impl::Deserializer) + sizeof(f) + payload_size, c->size )
245  << "Immediate buffer size to small to contain "
246  << sizeof(Grappa::impl::Deserializer) << "-byte deserializer + "
247  << sizeof(f) << "-byte lambda + "
248  << payload_size << " payload";
249 
250  *((void**)buf) = (void*) Grappa::impl::immediate_deserializer_with_payload<F>;
251  buf += sizeof(Grappa::impl::Deserializer);
252 
253  memcpy( buf, &f, sizeof(f) );
254  buf += sizeof(f);
255 
256  memcpy( buf, payload, payload_size );
257  buf += payload_size;
258 
259  post_send( c, dest, (buf - ((char*) c->buf)) );
260  }
