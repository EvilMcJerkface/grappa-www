31 /* Vector that exposes its storage array as read-only.
32  * increases in size as elements are added.
33  *
34  * Only supports insertions. Insertions allowed only
35  * while vector is in write only (WO) mode and
36  * getting the buffer pointer only allowed in read only mode
37  */
38

71  void insert( const T& v ) {
72  CHECK( mode == WO ) << "Must be in WO mode to insert";
73 
74  if ( nextIndex == size ) {
75  // expand the size of the buffer
76  size_t newsize = size * 2;
77  T * newbuf = static_cast<T*>(Grappa::impl::locale_shared_memory.allocate( newsize*sizeof(T) )); // new T[newsize]
78  memcpy( newbuf, buf, size*sizeof(T) );
79  Grappa::impl::locale_shared_memory.deallocate( buf ); // delete buf
80  size = newsize;
81  buf = newbuf;
82  }
83 
84  // store element and increment index
85  buf[nextIndex++] = v;
86  }
87 
88  GlobalAddress<const T> getReadBuffer() const {
89  CHECK( mode == RO ) << "Must be in RO mode to see buffer";
90  return make_global( const_cast<const T*>(buf) );
91  }
92 
93  size_t getLength() const {
94  return nextIndex;
95  }
