37  class CountingSemaphore {

39  inline void check( int64_t new_count ) const {
40  DLOG_ASSERT( ( -(1<<15) <= new_count ) &&
41  ( new_count < (1<<15) ) ) << "adjusting semaphore by too much";
42  }

70  // Increment semaphore. Wakes waiters
71  void increment( int64_t incr = 1 ) {
72  DVLOG(5) << this << "/" << Grappa::impl::global_scheduler.get_current_thread() << ": Incrementing " << count_ << " by " << incr;
73  check( count_ + incr ); // verify this is possible
74  count_ += incr; //
75  // TODO: is this a good choice?
76  // we need to do something intelligent to support decrements != 1
77  // the simplest correct thing is to wake all waiters,
78  // but this may not be efficient.
79  Grappa::broadcast( this );
80  }
81 
82  // Decrement semaphore. Blocks if decrement would leave semaphore < 0.
83  void decrement( int64_t decr = 1 ) {
84  DVLOG(5) << this << "/" << Grappa::impl::global_scheduler.get_current_thread() << ": Ready to decrement " << count_ << " by " << decr;
85  while( count_ - decr < 0 ) {
86  DVLOG(5) << this << "/" << Grappa::impl::global_scheduler.get_current_thread() << ": Blocking to decrement " << count_ << " by " << decr;
87  Grappa::wait( this );
88  DVLOG(5) << this << "/" << Grappa::impl::global_scheduler.get_current_thread() << ": Woken to decrement " << count_ << " by " << decr;
89  }
90  check( count_ - decr ); // ensure this adjustment is possible....
91  count_ -= decr;
92  }
93 
94  // Try to decrement semaphore without blocking. Returns true if it
95  // succeeds; false if it would block.
96  bool try_decrement( int64_t decr = 1 ) {
97  DVLOG(5) << this << "/" << Grappa::impl::global_scheduler.get_current_thread() << ": Trying to decrement " << count_ << " by " << decr;
98  if( count_ - decr < 0 ) {
99  DVLOG(5) << this << "/" << Grappa::impl::global_scheduler.get_current_thread() << ": Couldn't decrement " << count_ << " by " << decr;
100  return false;
101  }
102  check( count_ - decr ); // ensure this adjustment is possible....
103  count_ -= decr;
104  return true;
105  }
106 
107  inline int16_t get_value() const {
108  return count_;
109  }

111  };
