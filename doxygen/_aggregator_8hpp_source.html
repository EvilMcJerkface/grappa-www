422  inline bool poll() {
423  GRAPPA_FUNCTION_PROFILE( GRAPPA_COMM_GROUP );
424 #ifdef VTRACE_FULL
425  VT_TRACER("poll");
426 #endif
427  stats.record_poll();
428 
429  uint64_t beforePoll = aggregator_bundles_received_;
430  global_communicator.poll();
431  uint64_t afterPoll = aggregator_bundles_received_;
432  bool pollUseful = afterPoll > beforePoll;
433 
434 
435  uint64_t ts = get_timestamp();
436 
437  uint64_t beforeDeaggregate = aggregator_messages_deaggregated_;
438  deaggregate();
439  uint64_t afterDeaggregate = aggregator_messages_deaggregated_;
440  bool deagUseful = afterDeaggregate > beforeDeaggregate;
441 
442  // timestamp overflows are silently ignored.
443  // since it would take many many years to see one, I think that's okay for now.
444  int num_flushes = 0;
445  while( !least_recently_sent_.empty() &&
446  ((-least_recently_sent_.top_priority() + FLAGS_aggregator_autoflush_ticks) < ts) &&
447  ((FLAGS_aggregator_max_flush == 0) || (num_flushes < FLAGS_aggregator_max_flush)) ) {
448  stats.record_timeout();
449  DVLOG(5) << "timeout for node " << least_recently_sent_.top_key()
450  << ": inserted at " << -least_recently_sent_.top_priority()
451  << " autoflush_ticks " << FLAGS_aggregator_autoflush_ticks
452  << " (current ts " << ts << ")";
453  flush( least_recently_sent_.top_key() ); // send.
454  ++num_flushes;
455  }
456  if( num_flushes > 0 ) stats.record_multiflush();
457  bool flushUseful = num_flushes > 0;
458  previous_timestamp_ = ts;
459 
460  return flushUseful || deagUseful || pollUseful;
461  }

481  inline void aggregate( Core destination, AggregatorAMHandler fn_p,
482  const void * args, const size_t args_size,
483  const void * payload, const size_t payload_size ) {
484  GRAPPA_FUNCTION_PROFILE( GRAPPA_COMM_GROUP );
485 #ifdef VTRACE_FULL
486  VT_TRACER("aggregate");
487 #endif
488  CHECK( destination < max_nodes_ ) << "destination:" << destination << " max_nodes_:" << max_nodes_;
489  Core target = get_target_for_node( destination );
490  CHECK( target < max_nodes_ ) << "target:" << target << " max_nodes_:" << max_nodes_;
491 
492  // make sure arg struct and payload aren't too big.
493  // in the future, this would lead us down a separate code path for large messages.
494  // for now, fail.
495  size_t total_call_size = Grappa_sizeof_message( args, args_size, payload, payload_size );
496  DVLOG(5) << "aggregating " << total_call_size << " bytes to "
497  << destination << "(target " << target << ")";
498  CHECK( total_call_size < buffer_size_ ) << "payload_size( " << payload_size << " )"
499  << "+args_size( " << args_size << " )"
500  << "+header_size( " << sizeof( AggregatorGenericCallHeader ) << " )"
501  << "= " << total_call_size << " of max( " << buffer_size_ << " )";
502 
503  AggregatorGenericCallHeader header = { reinterpret_cast< uintptr_t >( fn_p ),
504  destination,
505  static_cast<uint16_t>(args_size),
506  static_cast<uint16_t>(payload_size)
507 #ifdef GRAPPA_TRACE
508  , global_communicator.mycore()
509 #endif
510 #ifdef VTRACE_FULL
511  , tag_
512 #endif
513  };
514 
515  if( FLAGS_aggregator_enable ) {
516  // does call fit in aggregation buffer?
517  if( !( buffers_[ target ].fits( total_call_size ) ) ) {
518  DVLOG(5) << "aggregating " << total_call_size << " bytes to "
519  << destination << "(target " << target
520  << "): didn't fit "
521  << "(current buffer position " << buffers_[ target ].current_position_
522  << ", next buffer position " << buffers_[ target ].current_position_ + total_call_size << ")";
523  // doesn't fit, so flush before inserting
524  stats.record_capacity_flush();
525  flush( target );
526  DCHECK( buffers_[ target ].fits( total_call_size ));
527  }
528 
529  // now call must fit, so just insert it
530  buffers_[ target ].insert( &header, sizeof( header ) );
531  buffers_[ target ].insert( args, args_size );
532  buffers_[ target ].insert( payload, payload_size );
533  stats.record_aggregation( total_call_size );
534 
535  } else {
536 
537  // aggregator is disabled; just send message
538  char * buf = raw_send_buffer_;
539  memcpy( buf, &header, sizeof( header ) );
540  buf += sizeof( header );
541  memcpy( buf, args, args_size );
542  buf += args_size;
543  memcpy( buf, payload, payload_size );
544  global_communicator.send_immediate_with_payload( target, [] (void * buf, int size) {
545  Aggregator_deaggregate_am( buf, size );
546  }, raw_send_buffer_, total_call_size );
547  }
548 
549  // trace fine-grain communication
550  if (FLAGS_record_grappa_events) {
551  // TODO: good candidate for TAU_CONTEXT_EVENT
552 #ifdef GRAPPA_TRACE
553  int fn_p_tag = aggregator_trace_tag( fn_p );
554  TAU_TRACE_SENDMSG(fn_p_tag, destination, args_size + payload_size );
555  // TODO: maybe add named communicators for separate function calls?
556 #endif
557 #ifdef VTRACE_FULL
558  VT_SEND( vt_agg_commid_, tag_, total_call_size );
559 #endif
560  }
561 
562  uint64_t ts = get_timestamp();
563  least_recently_sent_.update_or_insert( target, -ts );
564  previous_timestamp_ = ts;
565 #ifdef VTRACE_FULL
566  tag_ += global_communicator.mycore();
567 #endif
568  DVLOG(5) << "aggregated " << header;
569 }
