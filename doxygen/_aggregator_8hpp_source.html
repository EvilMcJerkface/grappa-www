438  inline bool poll() {
439  GRAPPA_FUNCTION_PROFILE( GRAPPA_COMM_GROUP );
440 #ifdef VTRACE_FULL
441  VT_TRACER("poll");
442 #endif
443  stats.record_poll();
444 
445  uint64_t beforePoll = aggregator_bundles_received_;
446  global_communicator.poll();
447  uint64_t afterPoll = aggregator_bundles_received_;
448  bool pollUseful = afterPoll > beforePoll;
449 
450 
451  uint64_t ts = get_timestamp();
452 
453  uint64_t beforeDeaggregate = aggregator_messages_deaggregated_;
454  deaggregate();
455  uint64_t afterDeaggregate = aggregator_messages_deaggregated_;
456  bool deagUseful = afterDeaggregate > beforeDeaggregate;
457 
458  // timestamp overflows are silently ignored.
459  // since it would take many many years to see one, I think that's okay for now.
460  int num_flushes = 0;
461  while( !least_recently_sent_.empty() &&
462  ((-least_recently_sent_.top_priority() + FLAGS_aggregator_autoflush_ticks) < ts) &&
463  ((FLAGS_aggregator_max_flush == 0) || (num_flushes < FLAGS_aggregator_max_flush)) ) {
464  stats.record_timeout();
465  DVLOG(5) << "timeout for node " << least_recently_sent_.top_key()
466  << ": inserted at " << -least_recently_sent_.top_priority()
467  << " autoflush_ticks " << FLAGS_aggregator_autoflush_ticks
468  << " (current ts " << ts << ")";
469  flush( least_recently_sent_.top_key() ); // send.
470  ++num_flushes;
471  }
472  if( num_flushes > 0 ) stats.record_multiflush();
473  bool flushUseful = num_flushes > 0;
474  previous_timestamp_ = ts;
475 
476  return flushUseful || deagUseful || pollUseful;
477  }

497  inline void aggregate( Core destination, AggregatorAMHandler fn_p,
498  const void * args, const size_t args_size,
499  const void * payload, const size_t payload_size ) {
500  GRAPPA_FUNCTION_PROFILE( GRAPPA_COMM_GROUP );
501 #ifdef VTRACE_FULL
502  VT_TRACER("aggregate");
503 #endif
504  CHECK( destination < max_nodes_ ) << "destination:" << destination << " max_nodes_:" << max_nodes_;
505  Core target = get_target_for_node( destination );
506  CHECK( target < max_nodes_ ) << "target:" << target << " max_nodes_:" << max_nodes_;
507 
508  // make sure arg struct and payload aren't too big.
509  // in the future, this would lead us down a separate code path for large messages.
510  // for now, fail.
511  size_t total_call_size = Grappa_sizeof_message( args, args_size, payload, payload_size );
512  DVLOG(5) << "aggregating " << total_call_size << " bytes to "
513  << destination << "(target " << target << ")";
514  CHECK( total_call_size < buffer_size_ ) << "payload_size( " << payload_size << " )"
515  << "+args_size( " << args_size << " )"
516  << "+header_size( " << sizeof( AggregatorGenericCallHeader ) << " )"
517  << "= " << total_call_size << " of max( " << buffer_size_ << " )";
518 
519  AggregatorGenericCallHeader header = { reinterpret_cast< uintptr_t >( fn_p ),
520  destination,
521  static_cast<uint16_t>(args_size),
522  static_cast<uint16_t>(payload_size)
523 #ifdef GRAPPA_TRACE
524  , global_communicator.mycore()
525 #endif
526 #ifdef VTRACE_FULL
527  , tag_
528 #endif
529  };
530 
531  if( FLAGS_aggregator_enable ) {
532  // does call fit in aggregation buffer?
533  if( !( buffers_[ target ].fits( total_call_size ) ) ) {
534  DVLOG(5) << "aggregating " << total_call_size << " bytes to "
535  << destination << "(target " << target
536  << "): didn't fit "
537  << "(current buffer position " << buffers_[ target ].current_position_
538  << ", next buffer position " << buffers_[ target ].current_position_ + total_call_size << ")";
539  // doesn't fit, so flush before inserting
540  stats.record_capacity_flush();
541  flush( target );
542  DCHECK( buffers_[ target ].fits( total_call_size ));
543  }
544 
545  // now call must fit, so just insert it
546  buffers_[ target ].insert( &header, sizeof( header ) );
547  buffers_[ target ].insert( args, args_size );
548  buffers_[ target ].insert( payload, payload_size );
549  stats.record_aggregation( total_call_size );
550 
551  } else {
552 
553  // aggregator is disabled; just send message
554  char * buf = raw_send_buffer_;
555  memcpy( buf, &header, sizeof( header ) );
556  buf += sizeof( header );
557  memcpy( buf, args, args_size );
558  buf += args_size;
559  memcpy( buf, payload, payload_size );
560  global_communicator.send( target,
561  aggregator_deaggregate_am_handle_,
562  raw_send_buffer_,
563  total_call_size );
564  }
565 
566  // trace fine-grain communication
567  if (FLAGS_record_grappa_events) {
568  // TODO: good candidate for TAU_CONTEXT_EVENT
569 #ifdef GRAPPA_TRACE
570  int fn_p_tag = aggregator_trace_tag( fn_p );
571  TAU_TRACE_SENDMSG(fn_p_tag, destination, args_size + payload_size );
572  // TODO: maybe add named communicators for separate function calls?
573 #endif
574 #ifdef VTRACE_FULL
575  VT_SEND( vt_agg_commid_, tag_, total_call_size );
576 #endif
577  }
578 
579  uint64_t ts = get_timestamp();
580  least_recently_sent_.update_or_insert( target, -ts );
581  previous_timestamp_ = ts;
582 #ifdef VTRACE_FULL
583  tag_ += global_communicator.mycore();
584 #endif
585  DVLOG(5) << "aggregated " << header;
586 }
