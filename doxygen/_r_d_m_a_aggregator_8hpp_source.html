678  void send_list_immediate( Grappa::impl::MessageBase * m ) {
679  app_messages_immediate++;
680 
681  const size_t size = m->serialized_size();
682 
683  CommunicatorContext * c = global_communicator.try_get_send_context();
684  if( !c ) {
685  global_communicator.garbage_collect();
686  c = global_communicator.try_get_send_context();
687  }
688  while( !c ) {
689  if( global_scheduler.in_no_switch_region() ) {
690  global_communicator.poll();
691  } else {
692  Grappa::yield();
693  }
694  global_communicator.garbage_collect();
695  c = global_communicator.try_get_send_context();
696  }
697 
698  Core dest = m->destination_;
699  char * buf = (char*) c->buf;
700  DCHECK_LE( size, c->size );
701 
702  // write deserializer pointer
703  typedef decltype(&deserialize_first_am) Deserializer;
704  Deserializer * dbuf = (Deserializer*) buf;
705  *dbuf = &deserialize_buffer_am;
706  buf = (char*) (dbuf + 1);
707 
708  size_t remaining = c->size - (buf - ((char*)c->buf));
709 
710  // serialize to buffer
711  Grappa::impl::MessageBase * tmp = m;
712  while( tmp != nullptr ) {
713  DVLOG(5) << __func__ << ": Serializing message from " << tmp;
714  char * end = aggregate_to_buffer( buf, &tmp, remaining );
715  DVLOG(5) << __func__ << ": After serializing, pointer was " << tmp;
716  DCHECK_EQ( end - buf, size ) << __func__ << ": Whoops! Aggregated message was too long to send as immediate";
717 
718  DVLOG(5) << __func__ << ": Sending " << end - buf
719  << " bytes of aggregated messages to " << dest;
720 
721  c->callback = NULL;
722  global_communicator.post_send( c, dest, end-((char*)c->buf) );
723  }
724  }
