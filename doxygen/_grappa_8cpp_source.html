116  void global_heap_init(size_t init_size) {
117  // by default, will allocate as much shared memory as it is
118  // possible to evenly split among the processors on a node
119  if (init_size != -1) {
120  impl::global_memory_size_bytes = init_size;
121  impl::global_bytes_per_core = init_size / cores();
122  impl::global_bytes_per_locale = init_size / locales();
123  return;
124  }
125 
126  // Decide how much memory we should allocate for global shared heap.
127  // this uses the locale shared size calculated in LocaleSharedMemory.cpp
128  auto sz = static_cast<int64_t>(FLAGS_locale_shared_size * FLAGS_global_heap_fraction);
129 
130  int64_t nnode = global_communicator.locales;
131  int64_t ppn = global_communicator.locale_cores;
132 
133  int64_t bytes_per_core = sz / ppn;
134  // round down to page size so we don't ask for too much
135  bytes_per_core &= ~( (1L << 12) - 1 );
136 
137  // be aware of hugepages
138  // Each core should ask for a multiple of 1GB hugepages
139  // and the whole node should ask for no more than the total pages available
140  if ( FLAGS_global_memory_use_hugepages ) {
141  int64_t pages_per_core = bytes_per_core / (1L << 30);
142  int64_t new_bpp = pages_per_core * (1L << 30);
143  if (new_bpp == 0) {
144  MASTER_ONLY VLOG(1) << "Allocating 1GB per core anyway.";
145  new_bpp = 1L << 30;
146  }
147  MASTER_ONLY VLOG_IF(1, bytes_per_core != new_bpp) << "With ppn=" << ppn << ", can only allocate "
148  << pages_per_core*ppn << " / " << FLAGS_node_memsize / (1L << 30) << " 1GB huge pages per node";
149  bytes_per_core = new_bpp;
150  }
151 
152  int64_t bytes = nnode * ppn * bytes_per_core;
153  int64_t bytes_per_node = ppn * bytes_per_core;
154  MASTER_ONLY DVLOG(2) << "bpp = " << bytes_per_core << ", bytes = " << bytes << ", bytes_per_node = " << bytes_per_node
155  << ", node_memsize = " << FLAGS_node_memsize << ", heap_size = " << sz;
156  MASTER_ONLY VLOG(2) << "nnode: " << nnode << ", ppn: " << ppn << ", iBs/node: " << log2((double)bytes_per_node) << ", total_iBs: " << log2((double)bytes);
157 
158  impl::global_memory_size_bytes = bytes;
159  impl::global_bytes_per_core = bytes_per_core;
160  impl::global_bytes_per_locale = bytes_per_node;
161  }

171 static void poller( Worker * me, void * args ) {
172  StateTimer::setThreadState( StateTimer::COMMUNICATION );
173  StateTimer::enterState_communication();
174  while( !Grappa_done() ) {
175  global_scheduler.stats.sample();
176 
177  Grappa::impl::poll();
178 
179  // check async. io completions
180  if (aio_completed_stack) {
181  // atomically grab the stack, replacing it with an empty stack again
182  IODescriptor * desc = __sync_lock_test_and_set(&aio_completed_stack, NULL);
183 
184  while (desc != NULL) {
185  desc->handle_completion();
186  IODescriptor * temp = desc->nextCompleted;
187  desc->nextCompleted = NULL;
188  desc = temp;
189  }
190  }
191 
192  Grappa::yield_periodic();
193  }
194  // cleanup stragglers on readyQ since I should be last to run;
195  // no one else matters.
196  // Tasks on task queues would be a programmer error
197  global_scheduler.shutdown_readyQ();
198  VLOG(5) << "polling Worker exiting";
199 
200  // master will be scheduled upon exit of poller thread
201 }

276 void adjust_footprints() {
277  auto locale_cores = global_communicator.locale_cores;
278  auto locale_total = FLAGS_locale_shared_size;
279 
280  // (FLAGS_locale_shared_size either set manually or computed from fraction of node_memsize)
281  auto locale_heap_bytes = static_cast<size_t>(FLAGS_locale_user_heap_fraction * (double)locale_total);
282  auto global_heap_bytes = Grappa::impl::global_memory_size_bytes;
283 
284  // memory left for Grappa components
285  auto grappa_bytes = (locale_total - global_heap_bytes - locale_heap_bytes) / locale_cores;
286  CHECK_GT(grappa_bytes, 0)
287  << "\nMust leave some memory for Grappa system components!\n"
288  << " - locale_heap_bytes: " << locale_heap_bytes << "\n"
289  << " - global_heap_bytes: " << global_heap_bytes << "\n"
290  << " - total: " << locale_total;
291 
292  auto total_footprint = []{
293  return global_communicator.estimate_footprint()
294  + global_rdma_aggregator.estimate_footprint()
295  + global_task_manager.estimate_footprint()
296  + SharedMessagePool::estimate_footprint();
297  };
298 
299  if (total_footprint() < grappa_bytes) return;
300 
301  // otherwise try to get all the grappa components to play along
302  long remaining = grappa_bytes;
303 
304  try {
305 
306  remaining -= SharedMessagePool::adjust_footprint(remaining / 4);
307  if (remaining < 0) throw "SharedMessagePool";
308  if (total_footprint() < grappa_bytes) throw true;
309 
310  remaining -= global_communicator.adjust_footprint(remaining / 3);
311  if (remaining < 0) throw "Communicator";
312  if (total_footprint() < grappa_bytes) throw true;
313 
314  remaining -= global_rdma_aggregator.adjust_footprint(remaining / 2);
315  if (remaining < 0) throw "RDMA Aggregator";
316  if (total_footprint() < grappa_bytes) throw true;
317 
318  remaining -= global_task_manager.adjust_footprint(remaining);
319  if (remaining < 0) throw "TaskManager";
320  if (total_footprint() < grappa_bytes) throw true;
321 
322  } catch (bool success) {
323 
324  MASTER_ONLY LOG(INFO) << "\nFootprint estimates: "
325  << "\n- locale_heap_bytes: " << locale_heap_bytes
326  << "\n- global_heap_bytes: " << global_heap_bytes
327  << "\n- total for Grappa: " << grappa_bytes
328  << "\n - global_communicator: " << global_communicator.estimate_footprint()
329  << "\n - global_rdma_aggregator: " << global_rdma_aggregator.estimate_footprint()
330  << "\n - global_task_manager: " << global_task_manager.estimate_footprint();
331 
332  } catch (char const* component) {
333  MASTER_ONLY LOG(ERROR)
334  << "\nUnable to fit Grappa components in memory. Failed at " << component
335  << "\n locale_heap_bytes: " << locale_heap_bytes
336  << "\n global_heap_bytes: " << global_heap_bytes
337  << "\n total for Grappa: " << grappa_bytes
338  << "\n global_communicator: " << global_communicator.estimate_footprint()
339  << "\n global_rdma_aggregator: " << global_rdma_aggregator.estimate_footprint()
340  << "\n global_task_manager: " << global_task_manager.estimate_footprint()
341  << "\n shared_message_pool: " << SharedMessagePool::estimate_footprint();
342  exit(1);
343  }
344 }

543 void Grappa_activate()
544 {

546 
547  locale_shared_memory.activate(); // do this before communicator

550  global_communicator.activate();

552 
553  global_task_manager.activate();

555

557 
558  // initializes system_wide global_memory pointer
559  global_memory = new GlobalMemory( Grappa::impl::global_memory_size_bytes );
560  auto heap_locale_shared_memory_allocated = locale_shared_memory.get_allocated();
561 
562  // fire up polling thread
563  global_scheduler.periodic( impl::worker_spawn( master_thread, &global_scheduler, &poller, NULL ) );
564  auto polling_locale_shared_memory_allocated = locale_shared_memory.get_allocated();
565 
566  global_rdma_aggregator.activate();
567  auto aggregator_locale_shared_memory_allocated = locale_shared_memory.get_allocated();
568 
569  SharedMessagePool::activate();
570  auto shared_pool_locale_shared_memory_allocated = locale_shared_memory.get_allocated();
571 
572  if (Grappa::mycore() == 0) {
573  double node_sz_gb = static_cast<double>(FLAGS_node_memsize) / (1L<<30);
574  double locale_sz_gb = static_cast<double>(FLAGS_locale_shared_size) / (1L<<30);
575  double locale_core_sz_gb = static_cast<double>(FLAGS_locale_shared_size) / Grappa::locale_cores() / (1L<<30);
576  double communicator_sz_gb = static_cast<double>( communicator_locale_shared_memory_allocated - base_locale_shared_memory_allocated ) / (1L<<30);
577  double tasks_sz_gb = static_cast<double>( tasks_locale_shared_memory_allocated - communicator_locale_shared_memory_allocated ) / (1L<<30);
578  double heap_sz_gb = static_cast<double>( heap_locale_shared_memory_allocated - tasks_locale_shared_memory_allocated ) / (1L<<30);
579  tasks_sz_gb += static_cast<double>( polling_locale_shared_memory_allocated - heap_locale_shared_memory_allocated ) / (1L<<30);
580  double aggregator_sz_gb = static_cast<double>( aggregator_locale_shared_memory_allocated - polling_locale_shared_memory_allocated ) / (1L<<30);
581  double shared_pool_sz_gb = static_cast<double>( shared_pool_locale_shared_memory_allocated - aggregator_locale_shared_memory_allocated ) / (1L<<30);
582  double shared_pool_max_sz_gb = static_cast<double>( FLAGS_shared_pool_max_size ) / (1L<<30);
583 
584  size_t free_sz = static_cast<double>(Grappa::impl::locale_shared_memory.get_free_memory());
585  double free_sz_gb = static_cast<double>(free_sz) / (1L<<30);
586  double free_core_sz_gb = static_cast<double>(free_sz) / Grappa::locale_cores() / (1L<<30);
587  VLOG(1) << "\n-------------------------\nShared memory breakdown:\n"
588  << " node total: " << node_sz_gb << " GB\n"
589  << " locale shared heap total: " << locale_sz_gb << " GB\n"
590  << " locale shared heap per core: " << locale_core_sz_gb << " GB\n"
591  << " communicator per core: " << communicator_sz_gb << " GB\n"
592  << " tasks per core: " << tasks_sz_gb << " GB\n"
