124 static void poller( Worker * me, void * args ) {
125  StateTimer::setThreadState( StateTimer::COMMUNICATION );
126  StateTimer::enterState_communication();
127  while( !Grappa_done() ) {
128  global_scheduler.stats.sample();
129 
130  Grappa::impl::poll();
131 
132  // poll global barrier
133  Grappa::impl::barrier_poll();
134 
135  // check async. io completions
136  if (aio_completed_stack) {
137  // atomically grab the stack, replacing it with an empty stack again
138  IODescriptor * desc = __sync_lock_test_and_set(&aio_completed_stack, NULL);
139 
140  while (desc != NULL) {
141  desc->handle_completion();
142  IODescriptor * temp = desc->nextCompleted;
143  desc->nextCompleted = NULL;
144  desc = temp;
145  }
146  }
147 
148  Grappa::yield_periodic();
149  }
150  // cleanup stragglers on readyQ since I should be last to run;
151  // no one else matters.
152  // Tasks on task queues would be a programmer error
153  global_scheduler.shutdown_readyQ();
154  VLOG(5) << "polling Worker exiting";
155 
156  // master will be scheduled upon exit of poller thread
157 }

202 void Grappa_init( int * argc_p, char ** argv_p[], int64_t global_memory_size_bytes)
203 {
204  // std::cerr << "Argc is " << *argc_p << std::endl;
205  // for( int i = 0; i < *argc_p; ++i ) {
206  // std::cerr << "Arg " << i << ": " << (*argv_p)[i] << std::endl;
207  // }
208 
209  // make sure gasnet is ready to backtrace
210  gasnett_backtrace_init( (*argv_p)[0] );
211 
212  // help generate unique profile filename
213  Grappa::impl::set_exe_name( (*argv_p)[0] );
214 
215  // parse command line flags
216  google::ParseCommandLineFlags(argc_p, argv_p, true);
217 
218  // activate logging
219  google::InitGoogleLogging( *argv_p[0] );
220  google::InstallFailureFunction( &Grappa::impl::failure_function );
221  google::OverrideDefaultSignalHandler( &gasnet_pause_sighandler );
222 
223  DVLOG(1) << "Initializing Grappa library....";
224 #ifdef HEAPCHECK_ENABLE
225  VLOG(1) << "heap check enabled";
226  Grappa_heapchecker = new HeapLeakChecker("Grappa");
227 #endif
228 
229  char * mem_reg_disabled = getenv("MV2_USE_LAZY_MEM_UNREGISTER");
230  if (mem_reg_disabled && strncmp(mem_reg_disabled,"0",1) == 0) {
231  VLOG(2) << "memory registration disabled";
232  }
233 
234  // how fast do we tick?
235  Grappa::force_tick();
236  Grappa::force_tick();
237  Grappa::Timestamp start_ts = Grappa::timestamp();
238  double start = Grappa::walltime();
239  // now go do other stuff for a while
240 
241  // set up stats dump signal handler
242  struct sigaction stats_dump_sa;
243  sigemptyset( &stats_dump_sa.sa_mask );
244  stats_dump_sa.sa_flags = 0;
245  stats_dump_sa.sa_handler = &stats_dump_sighandler;
246  CHECK_EQ( 0, sigaction( stats_dump_signal, &stats_dump_sa, 0 ) ) << "Stats dump signal handler installation failed.";
247  struct sigaction sigabrt_sa;
248  sigemptyset( &sigabrt_sa.sa_mask );
249  sigabrt_sa.sa_flags = 0;
250  sigabrt_sa.sa_handler = &gasnet_pause_sighandler;
251  CHECK_EQ( 0, sigaction( SIGABRT, &sigabrt_sa, 0 ) ) << "SIGABRT signal handler installation failed.";
252 
253  // Asynchronous IO
254  // initialize completed stack
255  aio_completed_stack = NULL;
256 
257  // handler
258 #ifdef AIO_SIGNAL
259  struct sigaction aio_sa;
260  aio_sa.sa_flags = SA_RESTART | SA_SIGINFO;
261  aio_sa.sa_sigaction = Grappa::impl::handle_async_io;
262  if (sigaction(AIO_SIGNAL, &aio_sa, NULL) == -1) {
263  fprintf(stderr, "Error setting up async io signal handler.\n");
264  exit(1);
265  }
266 #endif
267 
268  // initializes system_wide global_communicator
269  global_communicator.init( argc_p, argv_p );
270 
271  CHECK( global_communicator.locale_cores() <= MAX_CORES_PER_LOCALE );
272 
273  // initializes system_wide global_aggregator
274  global_aggregator.init();
275 
276  // set CPU affinity if requested
277 #ifdef CPU_SET
278  if( FLAGS_set_affinity ) {
279  char * localid_str = getenv("SLURM_LOCALID");
280  if( NULL != localid_str ) {
281  int localid = atoi( localid_str );
282  cpu_set_t mask;
283  CPU_ZERO( &mask );
284  CPU_SET( localid, &mask );
285  sched_setaffinity( 0, sizeof(mask), &mask );
286  }
287  }
288 #endif
289 
290  // initialize node shared memory
291  locale_shared_memory.init();
292 
293  // by default, will allocate as much shared memory as it is
294  // possible to evenly split among the processors on a node
295  if (global_memory_size_bytes == -1) {
296 
297  // Decide how much memory we should allocate for global shared heap.
298  // TODO: this should be a long literal
299  double shmmax_fraction = static_cast< double >( SHMMAX ) * FLAGS_global_heap_fraction;
300  int64_t shmmax_adjusted_floor = static_cast< int64_t >( shmmax_fraction );
301 
302  int64_t nnode = global_communicator.locales();
303  int64_t ppn = global_communicator.locale_cores();
304 
305  int64_t bytes_per_proc = shmmax_adjusted_floor / ppn;
306  // round down to page size so we don't ask for too much?
307  bytes_per_proc &= ~( (1L << 12) - 1 );
308 
309  // be aware of hugepages
310  // Each core should ask for a multiple of 1GB hugepages
311  // and the whole node should ask for no more than the total pages available
312  if ( FLAGS_global_memory_use_hugepages ) {
313  int64_t pages_per_proc = bytes_per_proc / (1L << 30);
314  int64_t new_bpp = pages_per_proc * (1L << 30);
315  if (new_bpp == 0) {
316  VLOG(1) << "Allocating 1GB per core anyway.";
317  new_bpp = 1L << 30;
318  }
319  VLOG_IF(1, bytes_per_proc != new_bpp) << "With ppn=" << ppn << ", can only allocate "
320  << pages_per_proc*ppn << " / " << SHMMAX / (1L << 30) << " 1GB huge pages per node";
321  bytes_per_proc = new_bpp;
322  }
323 
324  int64_t bytes = nnode * ppn * bytes_per_proc;
325  int64_t bytes_per_node = ppn * bytes_per_proc;
326  DVLOG(2) << "bpp = " << bytes_per_proc << ", bytes = " << bytes << ", bytes_per_node = " << bytes_per_node
327  << ", SHMMAX = " << SHMMAX << ", shmmax_adjusted_floor = " << shmmax_adjusted_floor;
328  VLOG(1) << "nnode: " << nnode << ", ppn: " << ppn << ", iBs/node: " << log2((double)bytes_per_node) << ", total_iBs: " << log2((double)bytes);
329  global_memory_size_bytes = bytes;
330 
331  Grappa::impl::global_memory_size_bytes = global_memory_size_bytes;
332  Grappa::impl::global_bytes_per_core = bytes_per_proc;
333  Grappa::impl::global_bytes_per_locale = bytes_per_node;
334  } else {
335  Grappa::impl::global_memory_size_bytes = global_memory_size_bytes;
336  Grappa::impl::global_bytes_per_core = global_memory_size_bytes / cores();
337  Grappa::impl::global_bytes_per_locale = global_memory_size_bytes / locales();
338  }
339 
340  VLOG(1) << "global_memory_size_bytes = " << Grappa::impl::global_memory_size_bytes;
341  VLOG(1) << "global_bytes_per_core = " << Grappa::impl::global_bytes_per_core;
342  VLOG(1) << "global_bytes_per_locale = " << Grappa::impl::global_bytes_per_locale;
343 
344  Grappa_done_flag = false;
345 
346  // process command line args for Tau
347  //TAU_INIT( argc_p, argv_p );
348 #ifdef GRAPPA_TRACE
349  TAU_PROFILE_SET_NODE(Grappa::mycore());
350 #endif
351 
352  //TODO: options for local stealing
353  node_neighbors = new Core[Grappa::cores()];
354  for ( Core nod=0; nod < Grappa::cores(); nod++ ) {
355  node_neighbors[nod] = nod;
356  }
357 
358  // start threading layer
359  master_thread = convert_to_master();
360  VLOG(1) << "Initializing tasking layer."
361  << " num_starting_workers=" << FLAGS_num_starting_workers;
362  global_task_manager.init( Grappa::mycore(), node_neighbors, Grappa::cores() ); //TODO: options for local stealing
363  global_scheduler.init( master_thread, &global_task_manager );
364 
365  // start RDMA Aggregator *after* threading layer
366  global_rdma_aggregator.init();
367 
368  // collect some stats on this job
369  Grappa::force_tick();
370  Grappa::force_tick();
371  Grappa::Timestamp end_ts = Grappa::timestamp();
372  double end = Grappa::walltime();
373  Grappa::tick_rate = (double) (end_ts - start_ts) / (end-start);
374 
375  char * jobid_str = getenv("SLURM_JOB_ID");
376  jobid = jobid_str ? atoi(jobid_str) : 0;
377  nodelist_str = getenv("SLURM_NODELIST");
378  if( NULL == nodelist_str ) nodelist_str = "undefined";
379 }

384 void Grappa_activate()
385 {
386  DVLOG(1) << "Activating Grappa library....";
387  global_communicator.activate();
388  locale_shared_memory.activate();
389  global_task_manager.activate();
390  Grappa::comm_barrier();
391 
392  // initializes system_wide global_memory pointer
393  global_memory = new GlobalMemory( Grappa::impl::global_memory_size_bytes );
394 
395  // fire up polling thread
396  global_scheduler.periodic( impl::worker_spawn( master_thread, &global_scheduler, &poller, NULL ) );
397 
398 
399  global_rdma_aggregator.activate();
400 
401  Grappa::init_shared_pool(); // (must be after locale-heap is initialized in RDMAAggregator)
402 
403  if (Grappa::mycore() == 0) {
404  size_t stack_sz = FLAGS_stack_size * FLAGS_num_starting_workers;
405  double stack_sz_gb = static_cast<double>(stack_sz) / (1L<<30);
406  double gheap_sz_gb = static_cast<double>(global_bytes_per_core) / (1L<<30);
407  size_t shpool_sz = FLAGS_shared_pool_max * FLAGS_shared_pool_size;
408  double shpool_sz_gb = static_cast<double>(shpool_sz) / (1L<<30);
409  size_t free_sz = Grappa::impl::locale_shared_memory.get_free_memory() / Grappa::locale_cores();
410  double free_sz_gb = static_cast<double>(free_sz) / (1L<<30);
411  VLOG(1) << "\n-------------------------\nShared memory breakdown:\n global heap: " << global_bytes_per_core << " (" << gheap_sz_gb << " GB)\n stacks: " << stack_sz << " (" << stack_sz_gb << " GB)\n rdma_aggregator: ??\n shared_message_pool: " << shpool_sz << " (" << shpool_sz_gb << " GB)\n free: " << free_sz << " (" << free_sz_gb << " GB)\n-------------------------";
412  }
413 
414  Grappa::comm_barrier();
415 }
