97  int64_t next_largest_power_of_2( int64_t v ) const {

99  v |= v >> 1;

131  void try_merge_buddy_recursive( ChunkMap::iterator cmit ) {
132  // compute address of buddy
133  intptr_t address = cmit->second.address;
134  intptr_t buddy_address = (address ^ cmit->second.size);
135  DVLOG(5) << cmit->second << " buddy address " << (void *) buddy_address;
136 
137  // does it exist?
138  ChunkMap::iterator buddy_iterator = chunks_.find( buddy_address );
139  if( buddy_iterator != chunks_.end() &&
140  buddy_iterator->second.size == cmit->second.size &&
141  buddy_iterator->second.in_use == false ) {
142  DVLOG(5) << "buddy found! address " << (void *) address << " buddy address " << (void *) buddy_address;
143 
144  // remove the higher-addressed chunk
145  ChunkMap::iterator higher_iterator = address < buddy_address ? buddy_iterator : cmit;
146  remove_from_free_list( higher_iterator );
147  chunks_.erase( higher_iterator );
148 
149  // keep the the lower-addressed chunk in the map:
150  // update its size and move it to the right free list
151  ChunkMap::iterator lower_iterator = address < buddy_address ? cmit : buddy_iterator;
152  remove_from_free_list( lower_iterator ); // should these be swapped? I think so.
153  lower_iterator->second.size *= 2;
154  add_to_free_list( lower_iterator );
155 
156  // see if we have more to merge
157  try_merge_buddy_recursive( lower_iterator );
158  }
159  }

165  Allocator( void * base, int64_t size )
166  : base_( reinterpret_cast< AllocatorAddress >( base ) )
167  , size_( size )
168  , chunks_( )
169  , free_lists_( )
170  {
171  // must pass a non-zero chunk to constructor
172  CHECK( size != 0 ) << "Must pass a non-zero chunk size to constructor";
173 
174  DVLOG(1) << "Allocator is responsible for addresses from " << base << " to " << (void*) ((char*) base + size);
175 
176  intptr_t offset = 0;
177 
178  // add chunks
179  while( size > 0 ) {
180  AllocatorAddress this_chunk_offset = offset;
181  size_t this_chunk_size = size;
182 
183  // is this chunk's size not a power of two?
184  if( ( size & (size - 1) ) != 0 ) {
185  this_chunk_size = next_largest_power_of_2( size / 2 );
186  //DVLOG(5) << "Not a power of two: adding chunk at " << this_chunk_offset << " with size " << this_chunk_size;
187  }
188 
189  ChunkMap::iterator it = add_to_chunk_map( AllocatorChunk( this_chunk_offset, this_chunk_size ) );
190  add_to_free_list( it );
191 
192  size -= this_chunk_size;
193  offset += this_chunk_size;
194  }
195  }

209  void * malloc( size_t size ) {
210  int64_t allocation_size = next_largest_power_of_2( size );
211 
212  // find a chunk large enough to start splitting.
213  FreeListMap::iterator flit = free_lists_.lower_bound( allocation_size );
214  if( flit == free_lists_.end() ) {
215  LOG(ERROR) << "Out of memory in the global heap: couldn't find a chunk of size " << allocation_size
216  << " to hold an allocation of " << size << " bytes. Can you increase --global_heap_fraction?";
217  // do this with an exception rather than CHECK_NE() so the test fixture can catch it.
218  throw Allocator::Exception();
219  }
220 
221  int64_t chunk_size = flit->first;
222  DVLOG(5) << "chunk_size is " << chunk_size;
223  DVLOG(5) << "free list size is " << flit->second.size();
224  ChunkMap::iterator cmit = flit->second.front(); // huh. this is broken.
225  // I think there's nothing there, even though there isn't supposed to be.
226  //DVLOG(5) << "cmit is " << (void*)cmit;
227  if( cmit == chunks_.end() ) DVLOG(5) << "nothing found. dump:" << *this;
228 
229  // subdivide chunk until we have what we need
230  while( chunk_size > allocation_size ) {
231 
232  // remove big chunk from free list
233  remove_from_free_list( cmit );
234 
235  // begin chopping chunk in half
236  chunk_size /= 2;
237  cmit->second.size = chunk_size;
238 
239  // add to new free list
240  add_to_free_list( cmit );
241 
242  // create new buddy chunk and add to chunk map and free list
243  intptr_t new_chunk_address = cmit->second.address + cmit->second.size;
244  cmit = add_to_chunk_map( AllocatorChunk( new_chunk_address, chunk_size ) );
245  add_to_free_list( cmit );
246 
247  }
248 
249  // finally we have a chunk of the right size
250  cmit->second.in_use = true;
251  remove_from_free_list( cmit );
252  return reinterpret_cast< void * >( cmit->second.address + base_ );
253  }
