80  void reset( ) {
81  DVLOG(5) << "In " << __PRETTY_FUNCTION__;
82  CHECK( !acquire_started_ || acquired_ ) << "inconsistent state for reset";
83  acquire_started_ = false;
84  acquired_ = false;
85  thread_ = NULL;
86  num_messages_ = 0;
87  response_count_ = 0;
88  expected_reply_payload_ = sizeof( T ) * *count_;
89  total_reply_payload_ = 0;
90  start_time_ = 0;
91  network_time_ = 0;
92  if( *count_ == 0 ) {
93  DVLOG(5) << "Zero-length acquire";
94  *pointer_ = NULL;
95  acquire_started_ = true;
96  acquired_ = true;
97  } else if( request_address_->is_2D() ) {
98  num_messages_ = 1;
99  if( request_address_->core() == Grappa::mycore() ) {
100  DVLOG(5) << "Short-circuiting to address " << request_address_->pointer();
101  *pointer_ = request_address_->pointer();
102  acquire_started_ = true;
103  acquired_ = true;
104  }
105  } else {
106  DVLOG(5) << "Straddle: block_max is " << (*request_address_ + *count_).block_max() ;
107  DVLOG(5) << ", request_address is " << *request_address_;
108  DVLOG(5) << ", sizeof(T) is " << sizeof(T);
109  DVLOG(5) << ", count is " << *count_;
110  DVLOG(5) << ", block_min is " << request_address_->block_min();
111 
112 
113  DVLOG(5) << "Straddle: address is " << *request_address_ ;
114  DVLOG(5) << ", address + count is " << *request_address_ + *count_;
115 
116  ptrdiff_t byte_diff = ( (*request_address_ + *count_ - 1).last_byte().block_max() -
117  request_address_->first_byte().block_min() );
118 
119  DVLOG(5) << "Straddle: address block max is " << request_address_->block_max();
120  DVLOG(5) << " address + count block max is " << (*request_address_ + *count_).block_max();
121  DVLOG(5) << " address block min " << request_address_->block_min();
122  DVLOG(5) << "Straddle: diff is " << byte_diff << " bs " << block_size;
123  num_messages_ = byte_diff / block_size;
124  }
125 
126  if( num_messages_ > 1 ) DVLOG(5) << "****************************** MULTI BLOCK CACHE REQUEST ******************************";
127 
128  DVLOG(5) << "In " << __PRETTY_FUNCTION__ << "; detecting straddle for sizeof(T):" << sizeof(T)
129  << " count:" << *count_
130  << " num_messages_:" << num_messages_
131  << " request_address:" << *request_address_;
132  }

134  void start_acquire() {
135  if( !acquire_started_ ) {
136 #ifdef VTRACE_FULL
137  VT_TRACER("incoherent start_acquire");
138 #endif
139  DVLOG(5) << "Worker " << Grappa::current_worker()
140  << " issuing acquire for " << *request_address_
141  << " * " << *count_ ;
142  acquire_started_ = true;
143  size_t total_bytes = *count_ * sizeof(T);
144 
145  // allocate enough requests/messages that we don't run out
146  size_t nmsg = total_bytes / block_size + 2;
147  size_t msg_size = sizeof(Grappa::Message<RequestArgs>);
148 
149  if (nmsg*msg_size < Grappa::current_worker()->stack_remaining()-8192) {
150  CHECK_LT(Grappa::current_worker()->stack_remaining(), STACK_SIZE);
151  // try to put message storage on stack if there's space
152  char msg_buf[nmsg*msg_size];
153  Grappa::MessagePool pool(msg_buf, sizeof(msg_buf));
154  do_acquire(pool);
155  } else {
156  // fall back on heap-allocating the storage
157  Grappa::MessagePool pool(nmsg*msg_size);
158  do_acquire(pool);
159  }
160  }
161  }

163  void do_acquire(Grappa::impl::MessagePoolBase& pool) {
164  size_t total_bytes = *count_ * sizeof(T);
165  RequestArgs args;
166  args.request_address = *request_address_;
167  DVLOG(5) << "Computing request_bytes from block_max " << request_address_->first_byte().block_max() << " and " << *request_address_;
168  args.reply_address = make_global( this );
169  args.offset = 0;
170 
171  for(size_t i = 0;
172  args.offset < total_bytes;
173  args.offset += args.request_bytes, i++) {
174 
175  args.request_bytes = args.request_address.first_byte().block_max() - args.request_address.first_byte();
176 
177 
178  if( args.request_bytes > total_bytes - args.offset ) {
179  args.request_bytes = total_bytes - args.offset;
180  }
181 
182  DVLOG(5) << "sending acquire request for " << args.request_bytes
183  << " of total bytes = " << *count_ * sizeof(T)
184  << " from " << args.request_address;
185 
186  pool.send_message(args.request_address.core(), [args]{
187  IAMetrics::count_acquire_ams( args.request_bytes );
188  DVLOG(5) << "Worker " << Grappa::current_worker()
189  << " received acquire request to " << args.request_address
190  << " size " << args.request_bytes
191  << " offset " << args.offset
192  << " reply to " << args.reply_address;
193 
194  DVLOG(5) << "Worker " << Grappa::current_worker()
195  << " sending acquire reply to " << args.reply_address
196  << " offset " << args.offset
197  << " request address " << args.request_address
198  << " payload address " << args.request_address.pointer()
199  << " payload size " << args.request_bytes;
200 
201  // note: this will read the payload *later* when the message is copied into the actual send buffer,
202  // should be okay because we're already assuming DRF, but something to watch out for
203  auto reply_address = args.reply_address;
204  auto offset = args.offset;
205 
206  Grappa::send_heap_message(args.reply_address.core(),
207  [reply_address, offset](void * payload, size_t payload_size) {
208  DVLOG(5) << "Worker " << Grappa::current_worker()
209  << " received acquire reply to " << reply_address
210  << " offset " << offset
211  << " payload size " << payload_size;
212  reply_address.pointer()->acquire_reply( offset, payload, payload_size);
213  },
214  args.request_address.pointer(), args.request_bytes
215  );
216 
217  DVLOG(5) << "Worker " << Grappa::current_worker()
218  << " sent acquire reply to " << args.reply_address
219  << " offset " << args.offset
220  << " request address " << args.request_address
221  << " payload address " << args.request_address.pointer()
222  << " payload size " << args.request_bytes;
223  });
224 
225  // TODO: change type so we don't screw with pointer like this
226  args.request_address = GlobalAddress<T>::Raw( args.request_address.raw_bits() + args.request_bytes );
227  }
228  DVLOG(5) << "acquire started for " << args.request_address;
229  }

231  void block_until_acquired() {
232  if( !acquired_ ) {
233  start_acquire();
234 #ifdef VTRACE_FULL
235  VT_TRACER("incoherent block_until_acquired");
236 #endif
237  DVLOG(5) << "Worker " << Grappa::current_worker()
238  << " ready to block on " << *request_address_
239  << " * " << *count_ ;
240  if( !acquired_ ) {
241  start_time_ = Grappa::timestamp();
242  } else {
243  start_time_ = 0;
244  }
245  while( !acquired_ ) {
246  DVLOG(5) << "Worker " << Grappa::current_worker()
247  << " blocking on " << *request_address_
248  << " * " << *count_ ;
249  if( !acquired_ ) {
250  thread_ = Grappa::current_worker();
251  Grappa::suspend();
252  thread_ = NULL;
253  }
254  DVLOG(5) << "Worker " << Grappa::current_worker()
255  << " woke up for " << *request_address_
256  << " * " << *count_ ;
257  }
258  IAMetrics::record_wakeup_latency( start_time_, network_time_ );
259  }

262  void acquire_reply( size_t offset, void * payload, size_t payload_size ) {
263  DVLOG(5) << "Worker " << Grappa::current_worker()
264  << " copying reply payload of " << payload_size
265  << " and waking Worker " << thread_;
266  memcpy( ((char*)(*pointer_)) + offset, payload, payload_size );
267  ++response_count_;
268  total_reply_payload_ += payload_size;
269  if ( response_count_ == num_messages_ ) {
270  DCHECK_EQ( total_reply_payload_, expected_reply_payload_ ) << "Got back the wrong amount of data "
271  << "(with base = " << *request_address_
272  << " and sizeof(T) = " << sizeof(T)
273  << " and count = " << *count_;
274  acquired_ = true;
275  if( thread_ != NULL ) {
276  Grappa::wake( thread_ );
277  }
278  if( start_time_ != 0 ) {
279  network_time_ = Grappa::timestamp();
280  IAMetrics::record_network_latency( start_time_ );
281  }
282  }
