153 void Aggregator::deaggregate( ) {
154  GRAPPA_FUNCTION_PROFILE( GRAPPA_COMM_GROUP );
155 #ifdef VTRACE_FULL
156  VT_TRACER("deaggregate");
157 #endif
158  StateTimer::enterState_deaggregation();
159  while( !received_AM_queue_.empty() ) {
160  DVLOG(5) << "deaggregating";
161  // TODO: too much copying
162  ReceivedAM amp = received_AM_queue_.front();
163 
164 #ifdef STL_DEBUG_ALLOCATOR
165  if( aggregator_access_control_active ) STLMemDebug::BaseAllocator::getMemMgr().setAccessMode(STLMemDebug::memReadWrite);
166 #endif
167  received_AM_queue_.pop();
168 #ifdef STL_DEBUG_ALLOCATOR
169  if( aggregator_access_control_active ) STLMemDebug::BaseAllocator::getMemMgr().setAccessMode(STLMemDebug::memReadOnly);
170 #endif
171 
172  DVLOG(5) << "deaggregating message of size " << amp.size_;
173  uintptr_t msg_base = reinterpret_cast< uintptr_t >( amp.buf_ );
174  for( unsigned int i = 0; i < amp.size_; ) {
175  AggregatorGenericCallHeader * header = reinterpret_cast< AggregatorGenericCallHeader * >( msg_base );
176  AggregatorAMHandler fp = reinterpret_cast< AggregatorAMHandler >( header->function_pointer );
177  void * args = reinterpret_cast< void * >( msg_base +
178  sizeof( AggregatorGenericCallHeader ) );
179  void * payload = reinterpret_cast< void * >( msg_base +
180  sizeof( AggregatorGenericCallHeader ) +
181  header->args_size );
182  if( header->destination == gasnet_mynode() ) { // for us?
183 
184  stats.record_deaggregation( sizeof( AggregatorGenericCallHeader ) + header->args_size + header->payload_size );
185  // trace fine-grain communication
186 #ifdef GRAPPA_TRACE
187  if (FLAGS_record_grappa_events) {
188  // TODO: good candidate for TAU_CONTEXT_EVENT
189  int fn_p_tag = aggregator_trace_tag( fp );
190  TAU_TRACE_RECVMSG(fn_p_tag, header->source, header->args_size + header->payload_size );
191  }
192 #endif
193 
194 #ifdef VTRACE_FULL
195  {
196  VT_RECV( vt_agg_commid_, header->tag, sizeof( AggregatorGenericCallHeader ) + header->args_size + header->payload_size );
197  }
198 #endif
199 
200 
201  DVLOG(5) << "calling " << *header
202  << " with args " << args
203  << " and payload " << payload;
204 
205  {
206  GRAPPA_PROFILE( deag_func_timer, "deaggregate execution", "", GRAPPA_USERAM_GROUP );
207  fp( args, header->args_size, payload, header->payload_size ); // execute
208  }
209  } else { // not for us, so forward towards destination
210  DVLOG(5) << "forwarding " << *header
211  << " with args " << args
212  << " and payload " << payload;
213  stats.record_forward( sizeof( AggregatorGenericCallHeader ) + header->args_size + header->payload_size );
214  aggregate( header->destination, fp, args, header->args_size, payload, header->payload_size );
215  }
216  i += sizeof( AggregatorGenericCallHeader ) + header->args_size + header->payload_size;
217  msg_base += sizeof( AggregatorGenericCallHeader ) + header->args_size + header->payload_size;
218  }
219  }
220 }
