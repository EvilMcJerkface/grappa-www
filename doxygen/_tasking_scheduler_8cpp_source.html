107 Worker * TaskingScheduler::thread_wait( void **result ) {
108  CHECK( current_thread == master ) << "only meant to be called by system Worker";
109 
110  Worker * next = nextCoroutine( false );
111  if (next == NULL) {
112  // no user threads remain
113  return NULL;
114  } else {
115  current_thread = next;
116 
117  Worker * died = (Worker *) thread_context_switch( master, next, NULL);
118 
119  // At the moment, we only return from a Worker in the case of death.
120 
121  if (result != NULL) {
122  void *retval = (void *)died->next;
123  *result = retval;
124  }
125  return died;
126  }
127 }

132 Worker * TaskingScheduler::getWorker () {
133  if (task_manager->available()) {
134  // check the pool of unassigned coroutines
135  Worker * result = unassignedQ.dequeue();
136  if (result != NULL) return result;
137 
138  // possibly spawn more coroutines
139  result = maybeSpawnCoroutines();
140  VLOG_IF(3, result==NULL) << "run out of coroutines";
141  return result;
142  } else {
143  return NULL;

153 void workerLoop ( Worker * me, void* args ) {
154  task_worker_args* wargs = (task_worker_args*) args;
155  TaskManager* tasks = wargs->tasks;
156  TaskingScheduler * sched = wargs->scheduler;
157 
158  sched->onWorkerStart();
159 
160  StateTimer::setThreadState( StateTimer::FINDWORK );
161  StateTimer::enterState_findwork();
162 
163  Task nextTask;
164 
165  while ( true ) {
166  // block until receive work or termination reached
167  if (!tasks->getWork(&nextTask)) break; // quitting time
168 
169  sched->num_active_tasks++;
170  StateTimer::setThreadState( StateTimer::USER );
171  StateTimer::enterState_user();
172  {
173  GRAPPA_PROFILE( exectimer, "user_execution", "", GRAPPA_USER_GROUP );
174  nextTask.execute();
175  }
176  StateTimer::setThreadState( StateTimer::FINDWORK );
177  sched->num_active_tasks--;
178 
179  sched->thread_yield( ); // yield to the scheduler
180  }
181 }

187 void TaskingScheduler::createWorkers( uint64_t num ) {
188  num_workers += num;
189  VLOG(5) << "spawning " << num << " workers; now there are " << num_workers;
190  for (uint64_t i=0; i<num; i++) {
191  // spawn a new worker Worker
192  Worker * t = impl::worker_spawn( current_thread, this, workerLoop, work_args);
193 
194  // place the Worker in the pool of idle workers
195  unassigned( t );
196  }
197  num_idle += num;
198 }
