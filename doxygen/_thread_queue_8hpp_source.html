88  void check_invariants() {
89  uint64_t max=std::numeric_limits<uint64_t>::min();
90  uint64_t min=std::numeric_limits<uint64_t>::max();
91  uint64_t sum=0;
92  for ( uint64_t i=0; i<num_queues; i++ ) {
93  max = _max<uint64_t>( max, queues[i].length() );
94  min = _min<uint64_t>( min, queues[i].length() );
95  sum += queues[i].length();
96  }
97  CHECK( static_cast<int64_t>(max) - static_cast<int64_t>(min) >= 0 );
98  CHECK( max - min <= 1 ) << "min=" << min << " max=" << max;
99  CHECK( sum == len );
100  }

121  void enqueue(Worker * t) {
122  queues[eq_index].enqueue( t );
123  eq_index = ((eq_index+1) == num_queues) ? 0 : eq_index+1;
124  len++;
125 
126 #ifdef DEBUG
127  check_invariants();
128 #endif
129  }

131  Worker * dequeue() {
132  Worker * result = queues[dq_index].dequeueLazy();
133  if ( result ) {
134  uint64_t t = dq_index;
135  dq_index = ((dq_index+1) == num_queues) ? 0 : dq_index+1;
136  len--;
137 
138 
139  // prefetch future stacks and thread objects
140  // let D = prefetch distance; N = num subqueues = D*2
141 
142 #ifdef WORKER_PREFETCH
143  __builtin_prefetch( result->next, // prefetch the next thread in the current subqueue (i.e. i or i+D*2 mod N)
144  1, // prefetch for RW
145  3 ); // prefetch with high temporal locality
146  //__builtin_prefetch( ((char*)(result->next))+64, 1, 3 );
147 #endif
148 
149  // now safe to NULL
150  result->next = NULL;
151 
152  uint64_t tstack = (t+(num_queues/2))%num_queues;//PERFORMANCE TODO: can optimize
153  Worker * tstack_worker = queues[tstack].front();
154  if ( tstack_worker ) {
155 #ifdef STACK_PREFETCH
156  __builtin_prefetch( tstack_worker->stack, // prefetch the stack that is prefetch distance away (i.e. i+D or i-1 mod N)
157  1, // prefetch for RW
158  3 ); // prefetch with high temporal locality
159  __builtin_prefetch( ((char*)(tstack_worker->stack))+64, 1, 3 );
160  __builtin_prefetch( ((char*)(tstack_worker->stack))+128, 1, 3 );
161  //__builtin_prefetch( (char*)(tstack_worker->stack)+128+64, 1, 3 );
162  //__builtin_prefetch( (char*)(tstack_worker->stack)+128+64+64, 1, 3 );
163 #endif
164  }
165 
166 #ifdef DEBUG
167  check_invariants();
168 #endif
169  return result;
170  } else {
171 #ifdef DEBUG
172  CHECK( dq_index == eq_index ) << "Empty queue invariant violated";
173  for ( uint64_t i=0; i<num_queues; i++ ) {
174  CHECK( queues[i].length() == 0 ) << "Empty queue invariant violated";
175  }
176  check_invariants();
177 #endif
178  return NULL;
179  }
180  }
