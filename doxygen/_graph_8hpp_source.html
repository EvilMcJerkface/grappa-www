608  template< typename V, typename E >
609  GlobalAddress<Graph<V,E>> Graph<V,E>::create(const TupleGraph& tg,
610  bool directed, bool solo_invalid) {
611  VLOG(1) << "Graph: " << (directed ? "directed" : "undirected");
612  double t;
613  auto g = symmetric_global_alloc<Graph>();
614 
615  // find nv
616  t = walltime();
617  forall(tg.edges, tg.nedge, [g](TupleGraph::Edge& e){
618  if (e.v0 > g->nv) { g->nv = e.v0; }
619  if (e.v1 > g->nv) { g->nv = e.v1; }
620  });
621  on_all_cores([g]{
622  g->nv = Grappa::allreduce<int64_t,collective_max>(g->nv) + 1;
623  });
624  VLOG(2) << "find_nv_time: " << walltime() - t;
625 
626  auto vs = global_alloc<Vertex>(g->nv);
627  auto self = g;
628  on_all_cores([g,vs]{
629  new (g.localize()) Graph(g, vs, g->nv);
630  for (Vertex& v : iterate_local(g->vs, g->nv)) {
631  new (&v) Vertex();
632  }
633 
634  #ifdef SMALL_GRAPH
635  // g->scratch = locale_alloc<int64_t>(g->nv);
636  if (locale_mycore() == 0) g->scratch = locale_alloc<int64_t>(g->nv);
637  barrier();
638  if (locale_mycore() == 0) {
639  memset(g->scratch, 0, sizeof(int64_t)*g->nv);
640  } else {
641  g->scratch = delegate::call(mylocale()*locale_cores(), [g]{ return g->scratch; });
642  }
643  VLOG(0) << "locale = " << mylocale() << ", scratch = " << g->scratch;
644  #endif
645  });
646  t = walltime();
647  // count the outgoing/undirected edges per vertex
648  forall(tg.edges, tg.nedge, [g,directed](TupleGraph::Edge& e){
649  CHECK_LT(e.v0, g->nv); CHECK_LT(e.v1, g->nv);
650  #ifdef SMALL_GRAPH
651  // g->scratch[e.v0]++;
652  // if (!directed) g->scratch[e.v1]++;
653  __sync_fetch_and_add(g->scratch+e.v0, 1);
654  if (!directed) __sync_fetch_and_add(g->scratch+e.v1, 1);
655  #else
656  auto count = [](GlobalAddress<Vertex> v){
657  delegate::call<SyncMode::Async>(v.core(), [v]{ v->local_sz++; });
658  };
659  count(g->vs+e.v0);
660  if (!directed) count(g->vs+e.v1);
661  #endif
662  });
663  VLOG(2) << "count_time: " << walltime() - t;
664 
665  #ifdef SMALL_GRAPH
666  t = walltime();
667  #ifdef USE_MPI3_COLLECTIVES
668  on_all_cores([g]{
669  MPI_Request r; int done;
670  MPI_Iallreduce(MPI_IN_PLACE, g->scratch, g->nv, MPI_INT64_T, MPI_SUM, global_communicator.grappa_comm, &r);
671  do {
672  MPI_Test( &r, &done, MPI_STATUS_IGNORE );
673  if(!done) { Grappa::yield(); }
674  } while(!done);
675  });
676  #else
677  on_all_cores([g]{ allreduce_inplace<int64_t,collective_add>(g->scratch, g->nv); });
678  #endif // USE_MPI3_COLLECTIVES
679  VLOG(2) << "allreduce_inplace_time: " << walltime() - t;
680  // on_all_cores([g]{ VLOG(5) << util::array_str("scratch", g->scratch, g->nv, 25); });
681  #endif // SMALL_GRAPH
682 
683  // allocate space for each vertex's adjacencies (+ duplicates)
684  forall(g->vs, g->nv, [g](int64_t i, Vertex& v) {
685  #ifdef SMALL_GRAPH
686  // adjust b/c allreduce didn't account for having 1 instance per locale
687  v.local_sz = g->scratch[i] / locale_cores();
688  #endif
689 
690  v.nadj = 0;
691  if (v.local_sz > 0) v.local_adj = new VertexID[v.local_sz];
692  });
693  VLOG(3) << "after adj allocs";
694 
695  // scatter
696  forall(tg.edges, tg.nedge, [g,directed](TupleGraph::Edge& e){
697  auto scatter = [g](int64_t vi, int64_t adj) {
698  auto vaddr = g->vs+vi;
699  delegate::call<SyncMode::Async>(vaddr.core(), [vaddr,adj]{
700  auto& v = *vaddr.pointer();
701  v.local_adj[v.nadj++] = adj;
702  });
703  };
704  scatter(e.v0, e.v1);
705  if (!directed) scatter(e.v1, e.v0);
706  });
707  VLOG(3) << "after scatter, nv = " << g->nv;
708 
709  // sort & de-dup
710  forall(g->vs, g->nv, [g](int64_t vi, Vertex& v){
711  CHECK_EQ(v.nadj, v.local_sz);
712  std::sort(v.local_adj, v.local_adj+v.nadj);
713 
714  int64_t tail = 0;
715  for (int64_t i=0; i<v.nadj; i++, tail++) {
716  v.local_adj[tail] = v.local_adj[i];
717  while (v.local_adj[tail] == v.local_adj[i+1]) i++;
718  }
719  v.nadj = tail;
720  // VLOG(0) << "<" << vi << ">" << util::array_str("", v.local_adj, v.nadj);
721  g->nadj_local += v.nadj;
722  });
723  VLOG(3) << "after sort";
724 
725  // compact
726  on_all_cores([g]{
727  #ifdef SMALL_GRAPH
728  if (locale_mycore() == 0) locale_free(g->scratch);
729  #endif
730 
731  VLOG(2) << "nadj_local = " << g->nadj_local;
732 
733  // allocate storage for local vertices' adjacencies
734  g->adj_buf = locale_alloc<VertexID>(g->nadj_local);
735  g->edge_storage = locale_alloc<EdgeState>(g->nadj_local);
736 
737  // default-initialize edges
738  // TODO: import edge info from TupleGraph
739  for (size_t i=0; i<g->nadj_local; i++) {
740  new (g->edge_storage+i) EdgeState();
741  }
742 
743  // compute total nadj
744  g->nadj = allreduce<int64_t,collective_add>(g->nadj_local);
745 
746  size_t offset = 0;
747  for (Vertex& v : iterate_local(g->vs, g->nv)) {
748  auto adj = g->adj_buf + offset;
749  Grappa::memcpy(adj, v.local_adj, v.nadj);
750  if (v.local_sz > 0) delete[] v.local_adj;
751  v.local_sz = v.nadj;
752  v.local_adj = adj;
753  v.local_edge_state = g->edge_storage+offset;
754  offset += v.nadj;
755  }
756  CHECK_EQ(offset, g->nadj_local);
757  });
758 
759  if (solo_invalid) {
760  // (note: this isn't necessary if we don't create vertices for those with no edges)
761  // find which are actually active (first, those with outgoing edges)
762  forall(g, [](Vertex& v){ v.valid = (v.nadj > 0); });
763  // then those with only incoming edges (reachable from at least one active vertex)
764  forall(g, [](Edge& e, Vertex& ve){ ve.valid = true; });
765  }
766  VLOG(1) << "-- vertices: " << g->nv;
767 
768  auto gsz = Vertex::global_heap_size()*g->nv
