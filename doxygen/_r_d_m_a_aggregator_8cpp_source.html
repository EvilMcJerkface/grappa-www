169  void RDMAAggregator::compute_route_map() {
170 
171  Core locale_first_core = Grappa::mylocale() * Grappa::locale_cores();
172  Core locale_last_core = Grappa::mylocale() * Grappa::locale_cores() + Grappa::locale_cores();
173 
174  // how many locales should seach core handle?
175  // (assume all locales have same core count)
176  // (make we have at least one locale per core)
177  // (round up)
178  Locale locales_per_core = 1 + ((Grappa::locales() - 1) / Grappa::locale_cores());
179 
180 
181  // initialize source cores
182  //source_core_for_locale_ = new Core[ Grappa::locales() ];
183  for( int i = 0; i < Grappa::locales(); ++i ) {
184  if( i == Grappa::mylocale() ) {
185  // locally, cores are responsible for their own messages
186  source_core_for_locale_[i] = -1;
187  } else {
188  // guess at correct source core on the node
189  Core offset = i / locales_per_core;
190  if( offset >= Grappa::locale_cores() ) { // if we guess too high
191  // give it to the core that would have been responsible for the local locale.
192  offset = Grappa::mylocale() / locales_per_core;
193  }
194  source_core_for_locale_[i] = Grappa::mylocale() * Grappa::locale_cores() + offset;
195  }
196  }
197 
198  // initialize destination cores.
199  // this should match up with source assignments.
200  //dest_core_for_locale_ = new Core[ Grappa::locales() ];
201  for( int i = 0; i < Grappa::locales(); ++i ) {
202  if( i == Grappa::mylocale() ) {
203  dest_core_for_locale_[i] = -1;
204  } else {
205  // guess at correct destination core on the node
206  Core offset = Grappa::mylocale() / locales_per_core;
207  if( offset >= Grappa::locale_cores() ) { // if we guess too high
208  // give it to the core that would have been responsible for
209  // the destination's locale.
210  offset = i / locales_per_core;
211  }
212  dest_core_for_locale_[i] = i * Grappa::locale_cores() + offset;
213 
214  }
215  }

218  DVLOG(2) << "From locale " << Grappa::mylocale() << " to locale " << i
