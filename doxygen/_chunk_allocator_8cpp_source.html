77 static void _aligned_allocator_append_chunk(struct aligned_allocator *aa,
78  size_t min_size) {
79  struct memory_chunk *new_chunk;
80 
81  chunkallocator_append++;
82 
83  new_chunk = Grappa::locale_alloc_aligned<struct memory_chunk>(CACHE_LINE_SIZE);
84  CHECK_NOTNULL(new_chunk);
85 
86  auto chunk_struct_size = sizeof( struct memory_chunk );
87  shared_pool_total_allocated += std::max( chunk_struct_size, (decltype(chunk_struct_size)) CACHE_LINE_SIZE );
88  chunkallocator_allocated += std::max( chunk_struct_size, (decltype(chunk_struct_size)) CACHE_LINE_SIZE );
89 
90  new_chunk->next = NULL;
91  new_chunk->chunk_size = MAX(min_size, aa->chunk_size) + aa->align_on;
92  new_chunk->chunk = Grappa::locale_alloc_aligned<char>(CACHE_LINE_SIZE, new_chunk->chunk_size);
93  CHECK_NOTNULL(new_chunk->chunk);
94 
95  auto chunk_size = sizeof(new_chunk->chunk_size);
96  shared_pool_total_allocated += std::max( new_chunk->chunk_size, (decltype(chunk_size)) CACHE_LINE_SIZE );
97 
98  // for timing only
99  memset(new_chunk->chunk, new_chunk->chunk_size, 0);
100 
101  if (!aa->first)
102  aa->last = aa->first = new_chunk;
103  else {
104  aa->last->next = new_chunk;
105  aa->last = new_chunk;
106  aa->last->next = NULL;
107  }
108  aa->last->offset = (size_t)((u_int64_t)ALIGN_UP(new_chunk->chunk, aa->align_on) -
109  (u_int64_t)new_chunk->chunk);
110 }

112 void *aligned_allocator_alloc(struct aligned_allocator *aa, size_t size) {
113  size = (size_t)ALIGN_UP(size, aa->align_on);
114  void *p;
115 
116  if (!aa->last)
117  _aligned_allocator_append_chunk(aa, size);
118  if ((aa->last->chunk_size - aa->last->offset) < size)
119  _aligned_allocator_append_chunk(aa, size);
120 
121  p = &aa->last->chunk[aa->last->offset];
122  aa->last->offset += size;
123 
124  return p;
125 }
