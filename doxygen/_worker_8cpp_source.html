124 void coro_spawn(Worker * me, Worker * c, coro_func f, size_t ssize) {
125  CHECK(c != NULL) << "Must provide a valid Worker";
126  c->running = 0;
127  c->suspended = 0;
128  c->idle = 0;
129 
130  // allocate stack and guard page
131  c->base = Grappa::impl::locale_shared_memory.allocate_aligned( ssize+4096*2, 4096 );
132  CHECK_NOTNULL( c->base );
133  c->ssize = ssize;
134 
135  // set stack pointer
136  c->stack = (char*) c->base + ssize + 4096 - current_stack_offset;
137 
138  // try to make sure we don't stuff all our stacks at the same cache index
139  const int num_offsets = 128;
140  const int cache_line_size = 64;
141  current_stack_offset += FLAGS_stack_offset;
142  current_stack_offset &= ((cache_line_size * num_offsets) - 1);
143 
144  c->tracking_prev = NULL;
145  c->tracking_next = NULL;
146 
147 #ifdef ENABLE_VALGRIND
148  c->valgrind_stack_id = VALGRIND_STACK_REGISTER( (char *) c->base + 4096, c->stack );
149 #endif
150 
151  // clear stack
152  memset(c->base, 0, ssize);
153 
154 #ifdef GUARD_PAGES_ON_STACK
155  // arm guard page
156  checked_mprotect( c->base, 4096, PROT_NONE );
157  checked_mprotect( (char*)c->base + ssize + 4096, 4096, PROT_NONE );
158 #endif
159 
160  // set up coroutine to be able to run next time we're switched in
161  makestack(&me->stack, &c->stack, f, c);
162 
163  insert_coro( c ); // insert into debugging list of coros
164 
165 #ifdef CORO_PROTECT_UNUSED_STACK
166  // disable writes to stack until we're swtiched in again.
167  checked_mprotect( (void*)((intptr_t)c->base + 4096), ssize, PROT_READ );
168  checked_mprotect( (void*)(c), 4096, PROT_READ );
169 #endif
170 
171  total_coros++;
172 }

198 void destroy_coro(Worker * c) {
199  total_coros--;
200 #ifdef ENABLE_VALGRIND
201  if( c->valgrind_stack_id != -1 ) {
202  VALGRIND_STACK_DEREGISTER( c->valgrind_stack_id );
203  }
204 #endif
205  if( c->base != NULL ) {
206  // disarm guard page
207  checked_mprotect( c->base, 4096, PROT_READ | PROT_WRITE );
208  checked_mprotect( (char*)c->base + c->ssize + 4096, 4096, PROT_READ | PROT_WRITE );
209 #ifdef CORO_PROTECT_UNUSED_STACK
210  // enable writes to stack so we can deallocate
211  checked_mprotect( (void*)((intptr_t)c->base + 4096), c->ssize, PROT_READ | PROT_WRITE );
212  checked_mprotect( (void*)(c), 4096, PROT_READ | PROT_WRITE );
213 #endif
214  remove_coro(c); // remove from debugging list of coros
215  Grappa::impl::locale_shared_memory.deallocate(c->base);
216  }
