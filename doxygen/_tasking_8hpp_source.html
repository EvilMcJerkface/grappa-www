75  template< typename T >
76  static void task_functor_proxy( uint64_t a0, uint64_t a1, uint64_t a2 ) {
77  uint64_t functor_storage[3] = { a0, a1, a2 };
78  T * tp = reinterpret_cast< T * >( &functor_storage[0] );

121  void privateTask( TF tf ) {
122  tasks_created++;
123  if( sizeof( tf ) > 24 ) { // if it's too big to fit in a task queue entry
124  DVLOG(4) << "Heap allocated task of size " << sizeof(tf);
125  tasks_heap_allocated++;

129  // heap-allocate copy of functor, passing ownership to spawned task

132  } else {
135  uint64_t * args = reinterpret_cast< uint64_t * >( &tf );

163  template < typename TF >
164  void spawn_worker( TF && tf ) {
165  TF * tp = new TF(tf);
166  void * vp = reinterpret_cast< void * >( tp );
167  Worker * th = impl::worker_spawn( Grappa::impl::global_scheduler.get_current_thread(), &Grappa::impl::global_scheduler,
168  Grappa::impl::worker_heapfunctor_proxy<TF>, vp );
169  Grappa::impl::global_scheduler.ready( th );
170  DVLOG(5) << __PRETTY_FUNCTION__ << " spawned Worker " << th;

183 template< typename FP >
184 void run(FP fp) {
185 #ifdef GRAPPA_TRACE
186  TAU_PROFILE("run_user_main()", "(user code entry)", TAU_USER);
187 #endif
188 #ifdef VTRACE
189  VT_TRACER("run_user_main()");
190 #endif
191 #ifdef VTRACE_SAMPLED
192  // this doesn't really add anything to the profiled trace
193  //Grappa_Grappa::impl::take_tracing_sample();
194 #endif
195 
196  if( global_communicator.mycore == 0 ) {
197  CHECK_EQ( Grappa::impl::global_scheduler.get_current_thread(), master_thread ); // this should only be run at the toplevel
198 
199  // create user_main as a private task
200  //Grappa_privateTask( &user_main_wrapper<A>, fp, args );
201  Grappa::spawn( [&fp] {
202  Grappa_global_queue_initialize();
203  fp();
204  Metrics::dump_stats_blob();
205  Grappa_end_tasks();
206  } );
207 
208  DVLOG(5) << "Spawned user_main";
209 
210  // spawn 1 extra worker that will take user_main
211  Grappa::impl::global_scheduler.createWorkers( 1 );
212  }
213 
214  // spawn starting number of worker coroutines
215  Grappa::impl::global_scheduler.createWorkers( FLAGS_num_starting_workers );
216  Grappa::impl::global_scheduler.allow_active_workers(-1); // allow all workers to be active
217 
218  StateTimer::init();
219 
220  // start the scheduler
221  Grappa::impl::global_scheduler.run( );
222 
223 #ifdef VTRACE_SAMPLED
224  // this doesn't really add anything to the profiled trace

227 }
