94  void sync() {
95  CompletionEvent ce(keys_to_insert.size()+lookups.size());
96  auto cea = make_global(&ce);
97 
98  for (auto& k : keys_to_insert) {
99  ++hashset_insert_msgs;
100  auto cell = owner->base+owner->computeIndex(k);
101  send_heap_message(cell.core(), [cell,k,cea]{
102  Cell * c = cell.localize();
103  bool found = false;
104  for (auto& e : c->entries) if (e.key == k) { found = true; break; }
105  if (!found) c->entries.emplace_back(k);
106  complete(cea);
107  });
108  }
109  for (auto& e : lookups) { auto k = e.first;
110  ++hashset_lookup_msgs;
111  auto re = e.second;
112  DVLOG(3) << "lookup " << k << " with re = " << re;
113  auto cell = owner->base+owner->computeIndex(k);
114 
115  send_heap_message(cell.core(), [cell,k,cea,re]{
116  Cell * c = cell.localize();
117  bool found = false;
118  for (auto& e : c->entries) if (e.key == k) { found = true; break; }
119 
120  send_heap_message(cea.core(), [cea,re,found]{
121  ResultEntry * r = re;
122  while (r != nullptr) {
123  r->result = found;
124  r = r->next;
125  }
126  complete(cea);
127  });
128  });
129  }
130  ce.wait();
131  }

204  // Inserts the key if not already in the set
205  //
206  // returns true if the set already contains the key
207  //
208  // synchronous operation
209  void insert( K key ) {
210  ++hashset_insert_ops;
211  if (FLAGS_flat_combining) {
212  proxy.combine([key](Proxy& p){ p.insert(key); return FCStatus::BLOCKED; });
213  } else {
214  ++hashset_insert_msgs;
215  delegate::call(base+computeIndex(key), [key](Cell * c) {
216  // find matching key in the list, if found, no insert necessary
217  for (auto& e : c->entries) if (e.key == key) return;
218 
219  // this is the first time the key has been seen so add it to the list
220  c->entries.emplace_back( key );
221  return;
222  });
223  }
