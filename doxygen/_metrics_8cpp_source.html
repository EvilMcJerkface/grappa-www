104  char * get_next_profiler_filename( ) {
105  // use Slurm environment variables if we can
106  // char * jobname = getenv("SLURM_JOB_NAME");
107  char * jobid = getenv("SLURM_JOB_ID");
108  char * procid = getenv("SLURM_PROCID");
109  if( /*jobname != NULL &&*/ jobid != NULL && procid != NULL ) {
110  sprintf( profiler_filename, "%s.%s.rank%s.phase%d.prof", "exe", jobid, procid, profiler_phase );
111  } else {
112  sprintf( profiler_filename, "%s.%d.pid%d.phase%d.prof", exe_name, time_for_profiler, getpid(), profiler_phase );
113  }
114  VLOG(3) << "profiler_filename = " << profiler_filename;
115  ++profiler_phase;
116  return profiler_filename;
117  }

125  void merge(MetricList& result) {
