193  void enroll(int64_t inc = 1) {
194  if (inc == 0) return;
195 
196  CHECK_GE(inc, 1);
197  count += inc;
198  DVLOG(5) << "enroll " << inc << " -> " << count << " gce("<<this<<")";
199 
200  // first one to have work here
201  if (count == inc) { // count[0 -> inc]
202  event_in_progress = true; // optimization to save checking in wait()
203  // cancel barrier
204  Core co = impl::call(master_core, [this] {
205  cores_out++;
206  return cores_out;
207  });
208  // first one to cancel barrier should make sure other cores are ready to wait
209  if (co == 1) { // cores_out[0 -> 1]
210  event_in_progress = true;
211  call_on_all_cores([this] {
212  event_in_progress = true;
213  });
214  CHECK(event_in_progress);
215  }
216  // block until cancelled
217  CHECK_GT(count, 0);
218  DVLOG(2) << "gce(" << this << " cores_out: " << co << ", count: " << count << ")";
219  }
220  }

226  void complete(int64_t dec = 1) {
227  count -= dec;
228  DVLOG(4) << "complete (" << count << ") -- gce(" << this << ")";
229 
230  // out of work here
231  if (count == 0) { // count[dec -> 0]
232  // enter cancellable barrier
233  send_heap_message(master_core, [this] {
234  cores_out--;
235  DVLOG(4) << "core entered barrier (cores_out:"<< cores_out <<")";
236 
237  // if all are in
238  if (cores_out == 0) { // cores_out[1 -> 0]
239  CHECK_EQ(count, 0);
240  // notify everyone to wake
241  for (Core c = 0; c < cores(); c++) {
242  send_heap_message(c, [this] {
243  CHECK_EQ(count, 0);
244  DVLOG(3) << "broadcast";
245  broadcast(&cv); // wake anyone who was waiting here
246  reset(); // reset, now anyone else calling `wait` should fall through
247  });
248  }
249  }
250  });
251  }
