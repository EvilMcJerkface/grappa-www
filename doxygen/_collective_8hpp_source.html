164  class InplaceReduction {
165  protected:
166  CompletionEvent * ce;
167  T * array;
168  Core elems_in = 0;

170  public:
173  void call_allreduce(T * in_array, size_t nelem) {
174  // setup everything (block to make sure HOME_CORE is done)
175  this->array = in_array;
176  this->nelem = nelem;
177 
178  size_t n_per_msg = MAX_MESSAGE_SIZE / sizeof(T);
179  size_t nmsg = nelem / n_per_msg + (nelem % n_per_msg ? 1 : 0);
180  auto nmsg_total = nmsg*(cores()-1);
181 
182  CompletionEvent local_ce;
183  this->ce = &local_ce;
184  this->ce->enroll( (mycore() == HOME_CORE) ? nmsg_total : nmsg );
185  barrier();
186 
187  if (mycore() != HOME_CORE) {
188  for (size_t k=0; k<nelem; k+=n_per_msg) {
189  size_t this_nelem = MIN(n_per_msg, nelem-k);
190 
191  // everyone sends their contribution to HOME_CORE, last one wakes HOME_CORE
192  send_heap_message(HOME_CORE, [this,k](void * payload, size_t payload_size) {
193  DCHECK(mycore() == HOME_CORE);
194 
195  auto in_array = static_cast<T*>(payload);
196  auto in_n = payload_size/sizeof(T);
197  auto total = this->array+k;
198 
199  for (size_t i=0; i<in_n; i++) {
200  total[i] = ReduceOp(total[i], in_array[i]);
201  }
202  DVLOG(3) << "incrementing HOME sem, now at " << ce->get_count();
203  this->ce->complete();
204  }, (void*)(in_array+k), sizeof(T)*this_nelem);
205  }
206 
207  DVLOG(3) << "about to block for " << nelem << " with sem == " << ce->get_count(); this->ce->wait();
208 
209  } else {
210  auto nmsg_total = nmsg*(cores()-1);
211 
212  // home core waits until woken by last received message from other cores
213  this->ce->wait();
214  DVLOG(3) << "woke with sem == " << ce->get_count();
215 
216  // send total to everyone else and wake them
217  char msg_buf[(cores()-1)*sizeof(PayloadMessage<std::function<void(decltype(this),size_t)>>)];
218  MessagePool pool(msg_buf, sizeof(msg_buf));
219  for (Core c = 0; c < cores(); c++) {
220  if (c != HOME_CORE) {
221  // send totals back to all the other cores
222  size_t n_per_msg = MAX_MESSAGE_SIZE / sizeof(T);
223  for (size_t k=0; k<nelem; k+=n_per_msg) {
224  size_t this_nelem = MIN(n_per_msg, nelem-k);
225  pool.send_message(c, [this,k](void * payload, size_t psz){
226  auto total_k = static_cast<T*>(payload);
227  auto in_n = psz / sizeof(T);
228  for (size_t i=0; i<in_n; i++) {
229  this->array[k+i] = total_k[i];
230  }
231  this->ce->complete();
232  DVLOG(3) << "incrementing sem, now at " << ce->get_count();
233  }, this->array+k, sizeof(T)*this_nelem);
234  }
235  }
236  }
237  // once all messages are sent, HOME_CORE's task continues
238  }
239  }
240  };
